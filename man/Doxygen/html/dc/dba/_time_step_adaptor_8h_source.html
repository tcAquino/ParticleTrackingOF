162  template <typename State, typename TimeGenerator, typename JumpGenerator>
163  double operator()
164  (State& state, TimeGenerator& time_generator, JumpGenerator& jump_generator) const
165  {
166  state.cell = _geometry.locator(state);
167  auto cell_id = state.cell;
168  if constexpr (check_if_outside)
169  if (outside<warn_if_outside>(state.cell, make_point(state.position), "Using nearest cell."))
170  cell_id = _geometry.locator.nearest_cell(state);
171  double cell_side = std::pow(_geometry.mesh().V()[cell_id]/_volume_factor,
172  1./Geometry::dim);
173  double local_advection_time = cell_side/Foam::mag(_velocity_field(state));
174  double local_diffusion_time = cell_side*cell_side/(2.*_params_transport.diff_coeff);
175 
176  double reaction_rate = 0.;
177  if constexpr (!std::is_same_v<SurfaceReaction, Reaction_DoNothing>)
178  for (auto face : _geometry.mesh().cells()[cell_id])
179  {
180  double rate = _surface_reaction.rate(face);
181  if (rate > reaction_rate)
182  reaction_rate = rate;
183  }
184  double local_reaction_time = _params_transport.diff_coeff/(reaction_rate*reaction_rate);
185 
186  double time_step = std::max(std::min({
187  _params_solvers.local_time_step_adv*local_advection_time,
188  _params_solvers.local_time_step_diff*local_diffusion_time,
189  _params_solvers.local_time_step_react*local_reaction_time }),
190  std::min({
191  _params_solvers.global_time_step_adv*_params_transport.advection_time,
192  _params_solvers.global_time_step_diff*_params_transport.diffusion_time,
193  _params_solvers.global_time_step_react*_params_reaction.reaction_time }));
194 
195  if constexpr (useful::has_time_step_setter<TimeGenerator>::value)
196  time_generator.time_step(time_step);
197  if constexpr (useful::has_time_step_setter<JumpGenerator>::value)
198  jump_generator.time_step(time_step);
199  if constexpr (useful::has_time_step_setter<SurfaceReaction>::value)
200  _surface_reaction.time_step(time_step);
201 
202  return time_step;
203  }

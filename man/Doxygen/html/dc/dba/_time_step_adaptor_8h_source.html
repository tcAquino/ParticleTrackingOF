112  TimeStepAdaptor_CellSize_SurfaceReaction(
113  Geometry const &geometry, VelocityField const &velocity_field,
114  SurfaceReaction &surface_reaction,
115  TransportParameters const &params_transport,
116  ReactionParameters const &params_reaction,
117  SolverParameters const &params_solvers)
118  : _geometry{geometry}, _velocity_field{velocity_field},
119  _surface_reaction{surface_reaction},
120  _params_transport{params_transport}, _params_reaction{params_reaction},
121  _params_solvers{params_solvers}, _volume_factor{1.} {
122  for (std::size_t dd = 2; dd-- > Geometry::dim;) {
123  Foam::vector direction = Foam::zero{};
124  direction[dd] = 1.;
125  std::vector<double> max;
126  std::vector<double> min;
127  for (auto const &patch : _geometry.mesh().boundary()) {
128  max.push_back(Foam::max(patch.Cf() & direction));
129  min.push_back(Foam::min(patch.Cf() & direction));
130  }
131  _volume_factor *= *std::max_element(max.begin(), max.end()) -
132  *std::min_element(min.begin(), min.end());
133  }

143  template <typename State, typename TimeGenerator, typename JumpGenerator>
144  double operator()(State &state, TimeGenerator &time_generator,
145  JumpGenerator &jump_generator) const {
146  auto cell_id = state.cell;
147  if constexpr (check_if_outside)
148  if (outside<warn_if_outside>(state.cell, make_point(state.position),
149  "Using nearest cell."))
150  cell_id = _geometry.locator.nearest_cell(state);
151  double cell_side = std::pow(_geometry.mesh().V()[cell_id] / _volume_factor,
152  1. / Geometry::dim);
153  double local_advection_time = cell_side / Foam::mag(_velocity_field(state));
154  double local_diffusion_time =
155  cell_side * cell_side / (2. * _params_transport.diff_coeff);
156 
157  double reaction_rate = 0.;
158  if constexpr (!std::is_same_v<SurfaceReaction, SurfaceReaction_DoNothing>)
159  for (auto face : _geometry.mesh().cells()[cell_id]) {
160  double rate = _surface_reaction.rate(face);
161  if (rate > reaction_rate)
162  reaction_rate = rate;
163  }
164  double local_reaction_time =
165  _params_transport.diff_coeff / (reaction_rate * reaction_rate);
166 
167  double time_step = std::max(
168  std::min({_params_solvers.local_time_step_adv * local_advection_time,
169  _params_solvers.local_time_step_diff * local_diffusion_time,
170  _params_solvers.local_time_step_react * local_reaction_time}),
171  std::min({_params_solvers.global_time_step_adv *
172  _params_transport.advection_time,
173  _params_solvers.global_time_step_diff *
174  _params_transport.diffusion_time,
175  _params_solvers.global_time_step_react *
176  _params_reaction.reaction_time}));
177 
178  if constexpr (meta::has_time_step_setter_v<TimeGenerator>)
179  time_generator.time_step(time_step);
180  if constexpr (meta::has_time_step_setter_v<JumpGenerator>)
181  jump_generator.time_step(time_step);
182  if constexpr (meta::has_time_step_setter_v<SurfaceReaction>)
183  _surface_reaction.time_step(time_step);
184 
185  return time_step;
186  }

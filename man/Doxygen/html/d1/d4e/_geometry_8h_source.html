405  Boundary_Periodic make_boundary_periodic
406  (Directories const& directories)
407  {
408  auto boundary_conditions =
409  get_boundary_conditions<dynamics>(directories);
410 
411  std::vector<std::vector<double>> boundaries_dim(dim);
412  for (auto const& bc : boundary_conditions)
413  if (bc.second == "periodic")
414  {
415  // Find average and variance of patch face centers.
416  auto const& patch_id =
417  mesh().boundaryMesh().findPatchID(bc.first, false);
418  auto const& patch = mesh().boundaryMesh()[patch_id];
419  auto face_start = patch.start();
420  Foam::point average_face_center = Foam::point::zero;
421  Foam::point variance_face_center = Foam::point::zero;
422  Foam::scalar total_area = 0.;
423  for (auto face = face_start;
424  face < face_start + patch.size(); ++face)
425  {
426  auto area = face_area(face, mesh());
427  auto weighted_center = area*face_center(face, mesh());
428  total_area += area;
429  average_face_center += weighted_center;
430  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
431  variance_face_center[dd] +=
432  weighted_center[dd]*weighted_center[dd];
433  }
434  average_face_center /= total_area;
435  variance_face_center /= total_area*total_area;
436  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
437  variance_face_center[dd] -=
438  average_face_center[dd]*average_face_center[dd];
439 
440  // Choose dimension with least variance.
441  auto it_min = std::min_element(variance_face_center.begin(),
442  variance_face_center.end());
443  std::size_t dd = std::distance(variance_face_center.begin(),
444  it_min);
445  boundaries_dim[dd].push_back(average_face_center[dd]);
446  }
447 
448  // Check which dimensions have periodic BCs.
449  // If a dimension has periodic BCs they must be exactly two.
450  std::vector<bool> dim_has_periodic_bcs(dim, 0);
451  for (std::size_t dd = 0; dd < dim; ++dd)
452  if (boundaries_dim[dd].size() != 0)
453  {
454  if (boundaries_dim[dd].size() != 2)
455  throw std::runtime_error{
456  "Extracted "
457  + std::to_string(boundaries_dim[dd].size())
458  + " periodic boundaries "
459  "along cartesian dimension "
460  + std::to_string(dd) };
461 
462  dim_has_periodic_bcs[dd] = 1;
463  }
464 
465  // Periodic conditions are assumed to be checked in order
466  // across Cartesian dimensions.
467  // If there are periodic BCs in all dimensions
468  // starting from 0 up to some dimension, add only those boundaries
469  // If there are gap dimensions between periodic dimensions,
470  // add {-inf, inf} boundaries.
471  bool found_nonperiodic = 0;
472  bool nonconsecutive_dims = 0;
473  for (bool periodic : dim_has_periodic_bcs)
474  {
475  if (found_nonperiodic && periodic)
476  {
477  nonconsecutive_dims = 1;
478  break;
479  }
480  if (!periodic)
481  found_nonperiodic = 1;
482  }
483  std::vector<std::pair<double, double>> boundaries;
484  for (std::size_t dd = 0; dd < dim; ++dd)
485  {
486  if (dim_has_periodic_bcs[dd])
487  {
488  boundaries.push_back({
489  std::min(boundaries_dim[dd][0],
490  boundaries_dim[dd][1]),
491  std::max(boundaries_dim[dd][0],
492  boundaries_dim[dd][1])
493  });
494  }
495  else if (nonconsecutive_dims)
496  {
497  boundaries.push_back({
498  -std::numeric_limits<double>::infinity(),
499  std::numeric_limits<double>::infinity()
500  });
501  }
502  }
503  return { boundaries };
504  }
505  };

244  template <typename IStream,
245  typename TransportParameters,
246  typename ReactionParameters>
247  void read_measure_spacing
248  (IStream& input,
249  TransportParameters const& params_transport,
250  ReactionParameters const& params_reaction)
251  {
252  useful::read_first_from_line(input, measure_spacing, comment_sequence);
253  useful::read_first_from_line(input, time_min, comment_sequence);
254  time_min *= time_unit_factor;
255  switch (MeasureSpacing::type(measure_spacing))
256  {
257  case MeasureSpacing::Type::step:
258  {
259  useful::read_first_from_line(input, time_increment, comment_sequence);
260  time_increment *= time_unit_factor;
261  time_max = std::numeric_limits<double>::infinity();
262  break;
263  }
264  case MeasureSpacing::Type::linear:
265  {
266  useful::read_first_from_line(input, time_max, comment_sequence);
267  time_max *= time_unit_factor;
268  std::size_t nr_measures;
269  useful::read_first_from_line(input, nr_measures, comment_sequence);
270  time_increment = (time_max - time_min)/(nr_measures - 1);
271  break;
272  }
273  case MeasureSpacing::Type::log:
274  {
275  useful::read_first_from_line(input, time_max, comment_sequence);
276  time_max *= time_unit_factor;
277  std::size_t nr_measures;
278  useful::read_first_from_line(input, nr_measures, comment_sequence);
279  time_increment = std::pow(time_max/time_min, 1./(nr_measures - 1));
280  break;
281  }
282  default:
283  throw std::runtime_error{
284  std::string("Measure spacing ")
285  + measure_spacing
286  + " not supported" };
287  }

291  template <typename IStream>
292  void read_measurement_types
293  (IStream& input)
294  {
295  std::string line;
296  while (std::getline(input, line))
297  {
298  if (line.find(comment_sequence) == 0)
299  continue;
300  useful::remove_carriage_return_in_place(line);
301  useful::clear_escape_in_place(line, comment_sequence);
302  auto split_line = useful::split(line);
303  std::size_t required_size = 1;
304  if (split_line.size() < required_size)
305  throw std::runtime_error{
306  "Could not parse measurement types" };
307  std::string const& name = split_line[0];
308  measurements.emplace_back(name);
309  if (name == "scalar_field" ||
310  name == "vector_field" ||
311  name == "tensor_field")
312  {
313  required_size = 2;
314  if (split_line.size() < required_size)
315  throw std::runtime_error{
316  std::string("Measurement type ") + name + ": "
317  "Field name not provided" };
318  measurements.back().field_name = split_line[1];
319  }
320  if (split_line.size() > required_size)
321  measurements.back().precision = std::stod(split_line.back());

97  // Get widths of bins having given values as midpoints
98  // and given minimum (left edge)
99  template <typename Value_type = double>
100  std::vector<Value_type> get_bin_widths
101  (std::vector<Value_type> const& values, Value_type minimum)
102  {
103  std::vector<Value_type> widths;
104  widths.reserve(values.size());
105  double left_edge = minimum;
106  for (auto const& val : values)
107  {
108  widths.push_back(2.*(val-left_edge));
109  if (widths.back() < 0.)
110  throw std::runtime_error{ "Bad minimum value, could not determine PDF bin widths" };
111  left_edge += widths.back();
112  }
113 
114  return widths;
115  }

153  // Split string into vector of strings
154  // Split at instances of delimeter
155  // Empty entries can be included or discarded
156  std::vector<std::string> split
157  (std::string const& string, std::string const& delimiter = " ",
158  bool empty_entries = false)
159  {
160  std::vector<std::string> tokens;
161 
162  for (std::size_t start = 0, end; start < string.length();
163  start = end+delimiter.length())
164  {
165  std::size_t position = string.find(delimiter, start);
166  end = position != std::string::npos? position : string.length();
167 
168  std::string token = string.substr(start, end-start);
169  if (empty_entries || !token.empty())
170  tokens.push_back(token);
171  }
172 
173  if (empty_entries &&
174  (string.empty() || endsWith(string, delimiter)))
175  tokens.push_back("");
176 
177  return tokens;
178  }

701  // Simple hash for a container by combining element hashes
702  template <typename Container>
703  struct hash_container
704  {
705  std::size_t operator()(Container const& container) const
706  {
707  using value_type = typename Container::value_type;
708  std::size_t hash_val = std::hash<value_type>()(container[ 0 ]);
709  for (size_t ii = 1; ii < container.size() - 1; ++ii)
710  {
711  hash_val = (hash_val ^ (std::hash<value_type>()(container[ii]) << 1)) >> 1;
712  }

714

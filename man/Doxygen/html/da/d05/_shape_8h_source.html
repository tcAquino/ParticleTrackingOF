167  // Reflect off sphere exterior (in any dimension)
168  // with old position outside and new position inside
169  // Put reflected position in reflected and contact point with sphere in contacg_point
170  // Note: it is safe to pass the same container for new and reflected quantities
171  template <typename Position, typename Sphere>
172  void reflectOffSphere_outsideToInside
173  (Position const& position_new, Position const& position_old,
174  Sphere const& sphere, Position& reflected, Position& contact_point)
175  {
176  // Vector from outside position to inside position
177  auto insideminusoutside = operation::minus(position_new, position_old);
178 
179  // Vector from outside position to sphere center
180  auto outsideminuscenter = operation::minus(position_old, sphere.center);
181 
182  // Find contact point with sphere
183  double coeff_a = operation::abs_sq(insideminusoutside);
184  double coeff_b = 2.*operation::dot(insideminusoutside, outsideminuscenter);
185  double coeff_c = operation::dot(outsideminuscenter, outsideminuscenter)
186  - sphere.radius*sphere.radius;
187  double fraction_to_intersection =
188  -(coeff_b + std::sqrt(coeff_b*coeff_b - 4.*coeff_a*coeff_c))/(2.*coeff_a);
189  operation::linearOp(fraction_to_intersection, insideminusoutside,
190  position_old, contact_point);
191 
192  // Vector from contact point to inside position
193  auto leftover = operation::times_scalar(1.-fraction_to_intersection,
194  insideminusoutside);
195 
196  // Compute unit normal to sphere surface at contact
197  auto normal = operation::minus(contact_point, sphere.center);
198  operation::div_scalar_InPlace(normal, sphere.radius);
199 
200  // Compute reflected position
201  operation::linearOp(-2.*operation::dot(leftover, normal), normal,
202  position_new, reflected);
203  }

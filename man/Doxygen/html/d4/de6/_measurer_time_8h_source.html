158  void operator()(double time) override {
159  _output << std::setw(_column_widths[0]) << time;
160  for (auto const &part : _subject.particles()) {
161  auto const &state = part.state_new();
162  auto cell = _locator(state);
163  _output << std::setw(_column_widths[1]) << state.tag;
164  if (!state.info.absorbed && part.state_old().time <= time &&
165  !outside(cell)) {
166  std::vector<int> in_region(_masks.size(), 0);
167  for (std::size_t ii = 0; ii < _masks.size(); ++ii)
168  if (_masks[ii].get()[cell] > _thresholds[ii])
169  in_region[ii] = 1;
170  if (std::any_of(in_region.begin(), in_region.end(),
171  [](int ii) { return ii > 0; })) {
172  useful::print(_output, state.position, _column_widths[2]);
173  _output << std::setw(_column_widths[3]) << state.mass;
174  useful::print(_output, in_region, _column_widths[4]);
175  }
176  }
177  }
178  _output << "\n";
179  }

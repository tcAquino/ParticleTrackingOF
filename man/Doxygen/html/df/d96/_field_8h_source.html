630 template <typename Field, typename Mesh>
631 auto magnitude_of_average(Field &field, Mesh const &mesh) {
632  Foam::scalar mesh_volume = Foam::sum(mesh.cellVolumes());
633  auto average_weighted_data = Foam::sum(field * mesh.cellVolumes());
634  return Foam::mag(average_weighted_data) / mesh_volume;
635 }
636 
639 template <typename Field, typename Mesh>
640 void rescale_to_average(Field &field, Mesh const &mesh, double average) {
641  field *= average / magnitude_of_average(field, mesh);
642 }

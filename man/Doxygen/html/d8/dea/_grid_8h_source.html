348  // Identify boundary positions along a dimension
349  // Values according to neighbors to "left" and "right"
350  // 0: void - void
351  // 1: solid - void
352  // 2: void - solid
353  // 3: solid - solid
354  template <typename Grid>
355  int void_type
356  (Grid const& grid,
357  std::vector<std::size_t> const& solid_idx,
358  std::size_t idx,
359  std::size_t dd)
360  {
361  std::size_t idx_left = grid.neighbor_cell_down(idx, dd);
362  std::size_t idx_right = grid.neighbor_cell_up(idx, dd);
363 
364  bool left_is_solid = useful::contains(solid_idx, idx_left);
365  bool right_is_solid = useful::contains(solid_idx, idx_right);
366 
367  if (!left_is_solid && !right_is_solid)
368  return 0;
369  if (left_is_solid && !right_is_solid)
370  return 1;
371  if (!left_is_solid && right_is_solid)
372  return 2;
373  return 3;
374  }

78  double box_size = boundaries.second - boundaries.first;

503 class Boundary_RadialReflecting_2d {
504 public:
505  const double radius;
506 
509  Boundary_RadialReflecting_2d(double radius) : radius{radius} {}
510 
516  template <typename Position>
517  bool out_of_bounds(Position const &position) const {
518  return op::abs_sq(position) > _radius_sq;
519  }
520 
526  template <typename State>
527  bool operator()(State &state, State const &state_old) const {
528  if (!out_of_bounds(state.position))
529  return false;
530 
531  auto position_old = state_old.position;
532  std::vector<double> jump = op::minus(state.position, position_old);
533  std::vector<double> tangent_at_contact(2);
534 
535  // Iterate reflection procedure until position is inside boundary
536  while (1) {
537  // Fraction alpha of jump up to the boundary
538  double pos_dot_jump = op::dot(position_old, jump);
539  double norm_sq_jump = op::abs_sq(jump);
540  double pos_sq_old = op::abs_sq(position_old);
541  double aux = pos_dot_jump / norm_sq_jump;
542  double alpha =
543  std::sqrt(aux * aux + (_radius_sq - pos_sq_old) / norm_sq_jump) - aux;
544 
545  // Avoid numerical issues
546  if (alpha <= 0. || std::isnan(alpha)) {
547  op::times_scalar_inplace(radius / op::abs(state.position),
548  state.position);
549  break;
550  }
551 
552  // Place old position at contact
553  for (std::size_t dd = 0; dd < 2; ++dd)
554  position_old[dd] += alpha * jump[dd];
555 
556  // Tangent to boundary at contact
557  tangent_at_contact[0] = position_old[1] / radius;
558  tangent_at_contact[1] = -position_old[0] / radius;
559 
560  // Angle between jump and tangent to boundary at contact
561  double tangent_dot_jump = op::dot(tangent_at_contact, jump);
562  double cos_contact_angle = tangent_dot_jump / std::sqrt(norm_sq_jump);
563  double contact_angle = std::acos(cos_contact_angle);
564 
565  // Outgoing jump at collision, rotate outer jump and flip the direction
566  op::rotate(jump, -2. * contact_angle);
567  op::times_scalar_inplace(1. - alpha, jump);
568 
569  // Place particle at reflected position
570  op::plus(position_old, jump, state.position);
571 
572  // If inside boundary, done
573  if (!out_of_bounds(state.position))
574  break;
575  }
576 
577  return true;
578  }
579 
580 private:
581  const double _radius_sq{radius * radius};
582 };

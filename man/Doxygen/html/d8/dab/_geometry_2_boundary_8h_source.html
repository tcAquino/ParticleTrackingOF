545  template <typename State>
546  bool operator()(State& state, State const& state_old) const
547  {
548  if (!out_of_bounds(state.position))
549  return false;
550 
551  auto position_old = state_old.position;
552  std::vector<double> jump = op::minus(state.position, position_old);
553  std::vector<double> tangent_at_contact(2);
554 
555  // Iterate reflection procedure until position is inside boundary
556  while (1)
557  {
558  // Fraction alpha of jump up to the boundary
559  double pos_dot_jump = op::dot(position_old, jump);
560  double norm_sq_jump = op::abs_sq(jump);
561  double pos_sq_old = op::abs_sq(position_old);
562  double aux = pos_dot_jump/norm_sq_jump;
563  double alpha = std::sqrt(aux*aux
564  +(_radius_sq-pos_sq_old)/norm_sq_jump)-aux;
565 
566  // Avoid numerical issues
567  if (alpha <= 0. || std::isnan(alpha))
568  {
569  op::times_scalar_inplace(radius/op::abs(state.position),
570  state.position);
571  break;
572  }
573 
574  // Place old position at contact
575  for (std::size_t dd = 0; dd < 2; ++dd)
576  position_old[dd] += alpha*jump[dd];
577 
578  // Tangent to boundary at contact
579  tangent_at_contact[0] = position_old[1]/radius;
580  tangent_at_contact[1] = -position_old[0]/radius;
581 
582  // Angle between jump and tangent to boundary at contact
583  double tangent_dot_jump = op::dot(tangent_at_contact, jump);
584  double cos_contact_angle = tangent_dot_jump/std::sqrt(norm_sq_jump);
585  double contact_angle = std::acos(cos_contact_angle);
586 
587  // Outgoing jump at collision, rotate outer jump and flip the direction
588  op::rotate(jump, -2.*contact_angle);
589  op::times_scalar_inplace(1.-alpha, jump);
590 
591  // Place particle at reflected position
592  op::plus(position_old, jump,
593  state.position);
594 
595  // If inside boundary, done
596  if(!out_of_bounds(state.position))
597  break;
598  }
599 
600  return true;
601  }

435  template <typename State>
436  bool operator()(State& state, State const& state_old) const
437  {
438  if (!outOfBounds(state.position))
439  return false;
440 
441  auto position_old = state_old.position;
442  std::vector<double> jump = operation::minus(state.position, position_old);
443  std::vector<double> tangent_at_contact(2);
444 
445  // Iterate reflection procedure until position is inside boundary
446  while (1)
447  {
448  // Fraction alpha of jump up to the boundary
449  double pos_dot_jump = operation::dot(position_old, jump);
450  double norm_sq_jump = operation::abs_sq(jump);
451  double pos_sq_old = operation::abs_sq(position_old);
452  double aux = pos_dot_jump/norm_sq_jump;
453  double alpha = std::sqrt(aux*aux
454  +(radius_sq-pos_sq_old)/norm_sq_jump)-aux;
455 
456  // Avoid numerical issues
457  if (alpha <= 0. || useful::isnan(alpha))
458  {
459  operation::times_scalar_InPlace(radius/operation::abs(state.position),
460  state.position);
461  break;
462  }
463 
464  // Place old position at contact
465  for (std::size_t dd = 0; dd < 2; ++dd)
466  position_old[dd] += alpha*jump[dd];
467 
468  // Tangent to boundary at contact
469  tangent_at_contact[0] = position_old[1]/radius;
470  tangent_at_contact[1] = -position_old[0]/radius;
471 
472  // Angle between jump and tangent to boundary at contact
473  double tangent_dot_jump = operation::dot(tangent_at_contact, jump);
474  double cos_contact_angle = tangent_dot_jump/std::sqrt(norm_sq_jump);
475  double contact_angle = std::acos(cos_contact_angle);
476 
477  // Outgoing jump at collision, rotate outer jump and flip the direction
478  operation::rotate(jump, -2.*contact_angle);
479  operation::times_scalar_InPlace(1.-alpha, jump);
480 
481  // Place particle at reflected position
482  operation::plus(position_old, jump,
483  state.position);
484 
485  // If inside boundary, done
486  if(!outOfBounds(state.position))
487  break;
488  }
489 
490  return true;
491  }

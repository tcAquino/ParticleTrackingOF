95 template <typename State>
96 void boundary_reflecting(State &state, Foam::point const &contact_point,
97  Foam::vector const &unit_normal) {
98  auto leftover_jump = make_point(state.position) - contact_point;
99 
100  // When a particle starts at a reflecting wall,
101  // it must not be reflected if it jumps towards the interior
102  // and not through the boundary
103  // This dot product can only be positive in that case
104  auto leftover_dot_normal = leftover_jump & unit_normal;
105  state.set_position(contact_point + leftover_jump +
106  (leftover_dot_normal < 0.
107  ? -2. * leftover_dot_normal * unit_normal
108  : Foam::zero{}));
109 }

129  std::string comment_sequence = "#";

131  auto file = useful::open_read(filename);
132  std::string line;
133  while (std::getline(file, line)) {
134  if (line.find(comment_sequence) == 0)
135  continue;
136  useful::remove_carriage_return_in_place(line);
137  useful::clear_escape_in_place(line, comment_sequence);

142  }
143  file.close();

312  template <typename State>
313  bool operator()(State &state, State const &state_old = {}) {
315  auto intersection = _locator.mesh_search().intersection(
316  make_point(state_old.position), make_point(state.position));
317 
318  // When the start point is on a cell face,
319  // sometimes the intersection with it is not found
320  // Avoid breaking by checking for intersections with a small
321  // backwards offset when the final state is out of bounds
322  if (!intersection.hit()) {
323  state.cell = _locator(state);
324  if (outside(state.cell)) {
325  intersection = _locator.mesh_search().intersection(
326  offset_backward_cell(make_point(state_old.position), state_old.cell,
327  make_point(state.position) -
328  make_point(state_old.position),
329  _locator),
330  make_point(state.position));
331  }
332  } else {
333  // Ignore new intersection if offset went beyond final point
334  if (((intersection.point() - make_point(state_old.position)) &
335  (make_point(state.position) - make_point(state_old.position))) < 0.)
336  intersection.setMiss();
337  }
338 
339  bool had_effect = 0;
340 
341  // While some patch is intersected
342  while (intersection.hit()) {
344  auto patch_id =
345  _locator.mesh().boundaryMesh().whichPatch(intersection.index());
346  auto patch = patch_name(patch_id);
347  auto type_name = _boundary_conditions.at(patch);
348 
349  // Apply the approriate boundary and store associated info
350  switch (_boundary_condition_types.type(type_name)) {
351  case BoundaryCondition::Type::reflecting: {
352  _store_info(state, state_old, intersection, _boundary_condition_types,
353  meta::Selector<BoundaryCondition::Type,
354  BoundaryCondition::Type::reflecting>{});
355  boundary_reflecting(state, intersection.point(),
356  reflection_normal(intersection.index()));
357  had_effect = 1;
358  break;
359  }
360  case BoundaryCondition::Type::reacting_reflecting: {
361  _store_info(state, state_old, intersection, _boundary_condition_types,
362  meta::Selector<BoundaryCondition::Type,
363  BoundaryCondition::Type::reflecting>{});
364  surface_reaction(state, state_old, intersection.index());
365  boundary_reflecting(state, intersection.point(),
366  reflection_normal(intersection.index()));
367  had_effect = 1;
368  break;
369  }
370  case BoundaryCondition::Type::periodic: {
371  _store_info(state, state_old, intersection, _boundary_condition_types,
372  meta::Selector<BoundaryCondition::Type,
373  BoundaryCondition::Type::periodic>{});
374  had_effect += _boundary_periodic(state, intersection);
375  break;
376  }
377  case BoundaryCondition::Type::absorbing: {
378  _store_info(state, state_old, intersection, _boundary_condition_types,
379  meta::Selector<BoundaryCondition::Type,
380  BoundaryCondition::Type::absorbing>{});
381  boundary_absorbing(state, intersection.point());
382  had_effect = 1;
383  break;
384  }
385  case BoundaryCondition::Type::custom: {
386  _store_info(state, state_old, intersection, _boundary_condition_types,
387  meta::Selector<BoundaryCondition::Type,
388  BoundaryCondition::Type::custom>{});
389  had_effect += _boundary_custom(state, state_old, intersection);
390  break;
391  }
392  case BoundaryCondition::Type::info: {
393  _store_info(state, state_old, intersection, _boundary_condition_types,
394  meta::Selector<BoundaryCondition::Type,
395  BoundaryCondition::Type::info>{});
396  break;
397  }
398  case BoundaryCondition::Type::empty: {
399  }
400  default:
401  throw std::runtime_error{"Boundary condition type " + type_name +
402  " not supported"};
403  }
404 
405  // Stop if right at the last intersection point
406  // This avoids numerical issues for some edge cases
407  if (make_point(state.position) == intersection.point())
408  break;
409 
410  // Find the next intersection with a boundary patch
411  auto old_intersection_point = intersection.point();
412  intersection = next_intersection(intersection, make_point(state.position),
413  state.cell);
414  // Ignore new intersection if offset went beyond final point
415  if (intersection.hit() &&
416  ((intersection.point() - old_intersection_point) &
417  (make_point(state.position) - old_intersection_point)) < 0.)
418  intersection.setMiss();
419  }
420 
421  return had_effect;
422  }

549 template <typename State, typename Intersection, typename Boundary,
550  typename Locator>
551 auto periodic_intersection(State state_outside, State const &state_image,
552  Intersection const &intersection,
553  Boundary const &boundary, Locator const &locator) {
554  // Compute displacement from intersection to outside position.
555  auto displacement = make_point(state_outside.position) - intersection.point();
556 
557  // Slight offset from intersection towards the outside point
558  // The goal is to find the periodic image
559  // of a point close to the intersection,
560  // so we can find its periodic image within the domain
561  // and then use it to find the periodic image of the intersection
562  // itself. Using a periodic image like this should work also for
563  // non-cartesian symmetry planes.
564  auto offset_pos = offset_forward_face(
565  intersection.point(), intersection.index(), displacement, locator);
566 
567  // Ensure offset position is closer to
568  // intersection than outside position
569  // If less than halfway
570  // (arbitrary point between the two; 0.25 of mag squared),
571  // set outside position to offset position
572  // otherwise, use the halfway point
573  if (Foam::magSqr(offset_pos - intersection.point()) <
574  0.25 * Foam::magSqr(displacement))
575  state_outside.set_position(offset_pos);
576  else
577  state_outside.set_position(intersection.point() + 0.5 * displacement);
578 
579  // Find the periodic image of the offset point
580  boundary(state_outside);
581 
582  // Find the periodic image of the intersection by
583  // looking from outside the domain towards the final periodic image
584  // Start at the outside point found by taking
585  return locator.mesh_search().intersection(
586  make_point(state_outside.position) -
587  2. * (make_point(state_image.position) -
588  make_point(state_outside.position)),
589  make_point(state_outside.position));
590 }

679 template <typename BCs, typename Mesh>
680 auto extract_cartesian_periodic_boundaries(BCs const &boundary_conditions,
681  Mesh const &mesh, std::size_t dim) {
682  std::vector<std::vector<double>> boundaries_dim(dim);
683  for (auto const &bc : boundary_conditions)
684  if (bc.second == "periodic") {
685  // Find average and variance of patch face centers.
686  auto const &patch_id = mesh.boundaryMesh().findPatchID(bc.first, false);
687  auto const &patch = mesh.boundaryMesh()[patch_id];
688  auto face_start = patch.start();
689  Foam::point average_face_center = Foam::point::zero;
690  Foam::point variance_face_center = Foam::point::zero;
691  Foam::scalar total_area = 0.;
692  for (auto face = face_start; face < face_start + patch.size(); ++face) {
693  auto area = face_area(face, mesh);
694  auto center = face_center(face, mesh);
695  auto weighted_center = area * center;
696  total_area += area;
697  average_face_center += weighted_center;
698  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
699  variance_face_center[dd] += weighted_center[dd] * center[dd];
700  }
701  average_face_center /= total_area;
702  variance_face_center /= total_area;
703  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
704  variance_face_center[dd] -=
705  average_face_center[dd] * average_face_center[dd];
706 
707  // Choose dimension with least variance.
708  auto it_min = std::min_element(variance_face_center.begin(),
709  variance_face_center.end());
710  std::size_t dd = std::distance(variance_face_center.begin(), it_min);
711  boundaries_dim[dd].push_back(average_face_center[dd]);
712  }
713 
714  // Check which dimensions have periodic BCs.
715  // If a dimension has periodic BCs they must be exactly two.
716  std::vector<bool> dim_has_periodic_bcs(dim, 0);
717  for (std::size_t dd = 0; dd < dim; ++dd)
718  if (boundaries_dim[dd].size() != 0) {
719  if (boundaries_dim[dd].size() != 2)
720  throw std::runtime_error{"Extracted " +
721  std::to_string(boundaries_dim[dd].size()) +
722  " periodic boundaries "
723  "along cartesian dimension " +
724  std::to_string(dd)};
725  dim_has_periodic_bcs[dd] = 1;
726  }
727 
728  // Periodic conditions are checked in order
729  // across Cartesian dimensions.
730  // If there are periodic BCs in all dimensions
731  // starting from 0 up to some dimension, add only those boundaries
732  // If there are gap dimensions between periodic dimensions,
733  // add {-inf, inf} boundaries.
734  bool found_nonperiodic = 0;
735  bool nonconsecutive_dims = 0;
736  for (bool periodic : dim_has_periodic_bcs) {
737  if (found_nonperiodic && periodic) {
738  nonconsecutive_dims = 1;
739  break;
740  }
741  if (!periodic)
742  found_nonperiodic = 1;
743  }
744  std::vector<std::pair<double, double>> boundaries;
745  boundaries.reserve(boundaries_dim.size());
746  for (std::size_t dd = 0; dd < boundaries_dim.size(); ++dd) {
747  if (dim_has_periodic_bcs[dd]) {
748  boundaries.push_back(
749  {std::min(boundaries_dim[dd][0], boundaries_dim[dd][1]),
750  std::max(boundaries_dim[dd][0], boundaries_dim[dd][1])});
751  } else if (nonconsecutive_dims) {
752  boundaries.push_back({-std::numeric_limits<double>::infinity(),
753  std::numeric_limits<double>::infinity()});
754  }
755  }
756 
757  return boundaries;
758 }

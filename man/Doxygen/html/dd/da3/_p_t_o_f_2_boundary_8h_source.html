104  template <typename State>
105  void boundary_reflecting
106  (State& state,
107  Foam::point const& contact_point,
108  Foam::vector const& unit_normal)
109  {
110  auto leftover_jump = make_point(state.position)
111  - contact_point;
112 
113  // When a particle starts at a reflecting wall,
114  // it must not be reflected if it jumps towards the interior
115  // and not through the boundary
116  // This dot product can only be positive in that case
117  auto leftover_dot_normal = leftover_jump&unit_normal;
118  state.set_position(contact_point
119  + leftover_jump
120  + (leftover_dot_normal < 0.
121  ? - 2.*leftover_dot_normal*unit_normal
122  : Foam::zero{}));
123  }

358  bool operator()
359  (State& state, State const& state_old = {})
360  {

363  _locator.mesh_search().intersection(make_point(state_old.position),
364  make_point(state.position));
365 
366  // When the start point is on a cell face,
367  // sometimes the intersection with it is not found
368  // Avoid breaking by checking for intersections with a small
369  // backwards offset when the final state is out of bounds
370  if (!intersection.hit())
371  {
372  state.cell = _locator(state);
373  if (outside(state.cell))
374  {
375  intersection =
376  _locator.mesh_search().intersection(offset_backward_cell(make_point(state_old.position),
377  state_old.cell,
378  make_point(state.position)
379  - make_point(state_old.position),
380  _locator),
381  make_point(state.position));
382  }
383  }
384  else
385  {
386  // Ignore new intersection if offset went beyond final point
387  if (((intersection.point() - make_point(state_old.position))
388  & (make_point(state.position) - make_point(state_old.position))) < 0.)
389  intersection.setMiss();

471  break;

484  // This avoids numerical issues for some edge cases
485  if (make_point(state.position) == intersection.point())
486  break;
487 
488  // Find the next intersection with a boundary patch
489  auto old_intersection_point = intersection.point();
490  intersection = next_intersection(intersection,
491  make_point(state.position),
492  state.cell);
493  // Ignore new intersection if offset went beyond final point
494  if (intersection.hit()
495  && ((intersection.point() - old_intersection_point)
496  & (make_point(state.position) - old_intersection_point)) < 0.)
497  intersection.setMiss();
498  }

636  template
637  <typename State, typename Intersection,
638  typename Boundary, typename Locator>
639  auto periodic_intersection
640  (State state_outside,
641  State const& state_image,
642  Intersection const& intersection,
643  Boundary const& boundary,
644  Locator const& locator)
645  {
646  // Compute displacement from intersection to outside position.
647  auto displacement = make_point(state_outside.position)-
648  intersection.point();
649 
650  // Slight offset from intersection towards the outside point
651  // The goal is to find the periodic image
652  // of a point close to the intersection,
653  // so we can find its periodic image within the domain
654  // and then use it to find the periodic image of the intersection
655  // itself. Using a periodic image like this should work also for
656  // non-cartesian symmetry planes.
657  auto offset_pos
658  = offset_forward_face(intersection.point(),
659  intersection.index(),
660  displacement,
661  locator);
662 
663  // Ensure offset position is closer to
664  // intersection than outside position
665  // If less than halfway
666  // (arbitrary point between the two; 0.25 of mag squared),
667  // set outside position to offset position
668  // otherwise, use the halfway point
669  if (Foam::magSqr(offset_pos - intersection.point())
670  < 0.25*Foam::magSqr(displacement))
671  state_outside.set_position(offset_pos);
672  else
673  state_outside.set_position(intersection.point()+
674  0.5*displacement);
675 
676  // Find the periodic image of the offset point
677  boundary(state_outside);
678 
679  // Find the periodic image of the intersection by
680  // looking from outside the domain towards the final periodic image
681  // Start at the outside point found by taking
682  return locator.mesh_search().
683  intersection(make_point(state_outside.position)
684  - 2.*(make_point(state_image.position)-
685  make_point(state_outside.position)),
686  make_point(state_outside.position));
687  }

790  template <typename BCs, typename Mesh>
791  auto extract_cartesian_periodic_boundaries
792  (BCs const& boundary_conditions,
793  Mesh const& mesh,
794  std::size_t dim)
795  {
796  std::vector<std::vector<double>> boundaries_dim(dim);
797  for (auto const& bc : boundary_conditions)
798  if (bc.second == "periodic")
799  {
800  // Find average and variance of patch face centers.
801  auto const& patch_id =
802  mesh.boundaryMesh().findPatchID(bc.first, false);
803  auto const& patch = mesh.boundaryMesh()[patch_id];
804  auto face_start = patch.start();
805  Foam::point average_face_center = Foam::point::zero;
806  Foam::point variance_face_center = Foam::point::zero;
807  Foam::scalar total_area = 0.;
808  for (auto face = face_start;
809  face < face_start + patch.size(); ++face)
810  {
811  auto area = face_area(face, mesh);
812  auto weighted_center = area*face_center(face, mesh);
813  total_area += area;
814  average_face_center += weighted_center;
815  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
816  variance_face_center[dd] +=
817  weighted_center[dd]*weighted_center[dd];
818  }
819  average_face_center /= total_area;
820  variance_face_center /= total_area*total_area;
821  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
822  variance_face_center[dd] -=
823  average_face_center[dd]*average_face_center[dd];
824 
825  // Choose dimension with least variance.
826  auto it_min = std::min_element(variance_face_center.begin(),
827  variance_face_center.end());
828  std::size_t dd = std::distance(variance_face_center.begin(),
829  it_min);
830  boundaries_dim[dd].push_back(average_face_center[dd]);
831  }
832 
833  // Check which dimensions have periodic BCs.
834  // If a dimension has periodic BCs they must be exactly two.
835  std::vector<bool> dim_has_periodic_bcs(dim, 0);
836  for (std::size_t dd = 0; dd < dim; ++dd)
837  if (boundaries_dim[dd].size() != 0)
838  {
839  if (boundaries_dim[dd].size() != 2)
840  throw std::runtime_error{
841  "Extracted "
842  + std::to_string(boundaries_dim[dd].size())
843  + " periodic boundaries "
844  "along cartesian dimension "
845  + std::to_string(dd) };
846 
847  dim_has_periodic_bcs[dd] = 1;
848  }
849 
850  // Periodic conditions are checked in order
851  // across Cartesian dimensions.
852  // If there are periodic BCs in all dimensions
853  // starting from 0 up to some dimension, add only those boundaries
854  // If there are gap dimensions between periodic dimensions,
855  // add {-inf, inf} boundaries.
856  bool found_nonperiodic = 0;
857  bool nonconsecutive_dims = 0;
858  for (bool periodic : dim_has_periodic_bcs)
859  {
860  if (found_nonperiodic && periodic)
861  {
862  nonconsecutive_dims = 1;
863  break;
864  }
865  if (!periodic)
866  found_nonperiodic = 1;
867  }
868  std::vector<std::pair<double, double>> boundaries;
869  for (std::size_t dd = 0; dd < dim; ++dd)
870  {
871  if (dim_has_periodic_bcs[dd])
872  {
873  boundaries.push_back({
874  std::min(boundaries_dim[dd][0],
875  boundaries_dim[dd][1]),
876  std::max(boundaries_dim[dd][0],
877  boundaries_dim[dd][1])
878  });
879  }
880  else if (nonconsecutive_dims)
881  {
882  boundaries.push_back({
883  -std::numeric_limits<double>::infinity(),
884  std::numeric_limits<double>::infinity()
885  });
886  }
887  }
888 
889  return boundaries;
890  }

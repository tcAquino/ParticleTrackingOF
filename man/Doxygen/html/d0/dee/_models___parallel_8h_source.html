47  template <typename Transport,
48  typename Solvers,
49  typename VelocityField,
50  typename Geometry,
51  typename Boundary,
52  typename ReactionParameters,
53  typename Reaction>
54  auto makeTransitions
55  (VelocityField const& velocity_field,
56  Geometry const& geometry,
57  Boundary& boundary,
58  typename Transport::Parameters const& params_transport,
59  ReactionParameters const& params_reaction,
60  typename Solvers::Parameters const& params_solvers,
61  Reaction const& reaction,
62  std::size_t num_threads)
63  {
64  using Transitions = decltype(ctrw::Transitions_CTRW_Transport_Reaction{
65  Transport::template makeTransitions<Solvers>(velocity_field,
66  geometry, boundary,
67  params_transport, params_reaction, params_solvers),
68  reaction });
69  std::vector<Transitions> transitions;
70  for (std::size_t thread = 0; thread < num_threads; ++thread)
71  transitions.emplace_back(Transport::template makeTransitions<Solvers>(velocity_field,
72  geometry,
73  boundary,
74  params_transport,
75  params_reaction,
76  params_solvers),
77  reaction);
78 
79  return transitions;

634  template <typename Locator>
635  auto cell_ids_region_cartesian
636  (std::vector<std::pair<double, double>> boundaries,
637  Locator const& locator)
638  {
639  // Identify degenerate dimensions
640  // (if some minimum and maximum boundary values are equal,
641  // the Cartesian region has lower dimension than the space)
642  std::vector<std::size_t> degenerate_dimensions;
643  std::vector<std::size_t> non_degenerate_dimensions;
644  for (std::size_t dd = 0; dd < boundaries.size(); ++dd)
645  if (boundaries[dd].first == boundaries[dd].second)
646  degenerate_dimensions.push_back(dd);
647  else
648  non_degenerate_dimensions.push_back(dd);
649 
650  auto const& mesh = locator.mesh();
651 
652  std::set<Foam::label> cell_ids;
653  for (Foam::label cc = 0; cc < mesh.nCells(); ++cc)
654  {
655  auto center = cell_center(cc, mesh);
656 
657  bool cell_is_within_non_degenerate_boundaries = 1;
658  for (auto dd : non_degenerate_dimensions)
659  {
660  if (center[dd] < boundaries[dd].first ||
661  center[dd] > boundaries[dd].second)
662  {
663  cell_is_within_non_degenerate_boundaries = 0;
664  break;
665  }
666  }
667  if (!cell_is_within_non_degenerate_boundaries)
668  continue;
669  if (degenerate_dimensions.size() == 0)
670  cell_ids.insert(cc);
671 
672  // Consider a position equal to the cell center of the candidate cell
673  // but with components along the degenerate dimension
674  // equal to the prescribed value. If it is in the
675  // mesh, include it in the region
676  for (auto dd : degenerate_dimensions)
677  center[dd] = degenerate_dimensions[dd];
678  auto cell_id = mesh.findCell(center);
679  if (!outside(cell_id))
680  cell_ids.insert(cell_id);
681  }
682 
683  std::vector<Foam::label> cells(cell_ids.begin(), cell_ids.end());
684  return cells;

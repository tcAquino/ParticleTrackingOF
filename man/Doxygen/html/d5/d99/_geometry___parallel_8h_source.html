382  Boundary_Periodic makeboundary_periodic
383  (Directories const& directories)
384  {
385  auto boundary_conditions = get_boundary_conditions<dynamics>(directories);
386 
387  std::vector<std::vector<double>> boundaries_dim(dim);
388  for (auto const& bc : boundary_conditions)
389  if (bc.second == "periodic")
390  {
391  // Find average and variance of patch face centers.
392  auto const& patch_id =
393  mesh().boundaryMesh().findPatchID(bc.first, false);
394  auto const& patch = mesh().boundaryMesh()[patch_id];
395  auto face_start = patch.start();
396  Foam::point average_face_center = Foam::point::zero;
397  Foam::point variance_face_center = Foam::point::zero;
398  Foam::scalar total_area = 0.;
399  for (auto face = face_start;
400  face < face_start + patch.size(); ++face)
401  {
402  auto area = face_area(face, mesh());
403  auto weighted_center = area*face_center(face, mesh());
404  total_area += area;
405  average_face_center += weighted_center;
406  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
407  variance_face_center[dd] +=
408  weighted_center[dd]*weighted_center[dd];
409  }
410  average_face_center /= total_area;
411  variance_face_center /= total_area*total_area;
412  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
413  variance_face_center[dd] -=
414  average_face_center[dd]*average_face_center[dd];
415 
416  // Choose dimension with least variance.
417  auto it_min = std::min_element(variance_face_center.begin(),
418  variance_face_center.end());
419  std::size_t dd = std::distance(variance_face_center.begin(),
420  it_min);
421  boundaries_dim[dd].push_back(average_face_center[dd]);
422  }
423 
424  // Check which dimensions have periodic BCs.
425  // If a dimension has periodic BCs they must be exactly two.
426  std::vector<bool> dim_has_periodic_bcs(dim, 0);
427  for (std::size_t dd = 0; dd < dim; ++dd)
428  if (boundaries_dim[dd].size() != 0)
429  {
430  if (boundaries_dim[dd].size() != 2)
431  throw std::runtime_error{
432  "Extracted "
433  + std::to_string(boundaries_dim[dd].size())
434  + " periodic boundaries "
435  "along cartesian dimension "
436  + std::to_string(dd) };
437 
438  dim_has_periodic_bcs[dd] = 1;
439  }
440 
441  // Periodic conditions are assumed to be checked in order
442  // across Cartesian dimensions.
443  // If there are periodic BCs in all dimensions
444  // starting from 0 up to some dimension, add only those boundaries
445  // If there are gap dimensions between periodic dimensions,
446  // add {-inf, inf} boundaries.
447  bool found_nonperiodic = 0;
448  bool nonconsecutive_dims = 0;
449  for (bool periodic : dim_has_periodic_bcs)
450  {
451  if (found_nonperiodic && periodic)
452  {
453  nonconsecutive_dims = 1;
454  break;
455  }
456  if (!periodic)
457  found_nonperiodic = 1;
458  }
459  std::vector<std::pair<double, double>> boundaries;
460  for (std::size_t dd = 0; dd < dim; ++dd)
461  {
462  if (dim_has_periodic_bcs[dd])
463  {
464  boundaries.push_back({
465  std::min(boundaries_dim[dd][0],
466  boundaries_dim[dd][1]),
467  std::max(boundaries_dim[dd][0],
468  boundaries_dim[dd][1])
469  });
470  }
471  else if (nonconsecutive_dims)
472  {
473  boundaries.push_back({
474  -std::numeric_limits<double>::infinity(),
475  std::numeric_limits<double>::infinity()
476  });
477  }
478  }
479  return { boundaries };
480  }

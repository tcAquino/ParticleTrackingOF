369  Boundary_Periodic makeboundary_periodic
370  (Directories const& directories)
371  {
372  auto boundary_conditions =
373  get_boundary_conditions<dynamics>(directories);
374 
375  std::vector<std::vector<double>> boundaries_dim(dim);
376  for (auto const& bc : boundary_conditions)
377  if (bc.second == "periodic")
378  {
379  // Find average and variance of patch face centers.
380  auto const& patch_id =
381  mesh().boundaryMesh().findPatchID(bc.first, false);
382  auto const& patch = mesh().boundaryMesh()[patch_id];
383  auto face_start = patch.start();
384  Foam::point average_face_center = Foam::point::zero;
385  Foam::point variance_face_center = Foam::point::zero;
386  Foam::scalar total_area = 0.;
387  for (auto face = face_start;
388  face < face_start + patch.size(); ++face)
389  {
390  auto area = face_area(face, mesh());
391  auto weighted_center = area*face_center(face, mesh());
392  total_area += area;
393  average_face_center += weighted_center;
394  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
395  variance_face_center[dd] +=
396  weighted_center[dd]*weighted_center[dd];
397  }
398  average_face_center /= total_area;
399  variance_face_center /= total_area*total_area;
400  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
401  variance_face_center[dd] -=
402  average_face_center[dd]*average_face_center[dd];
403 
404  // Choose dimension with least variance.
405  auto it_min = std::min_element(variance_face_center.begin(),
406  variance_face_center.end());
407  std::size_t dd = std::distance(variance_face_center.begin(),
408  it_min);
409  boundaries_dim[dd].push_back(average_face_center[dd]);
410  }
411 
412  // Check which dimensions have periodic BCs.
413  // If a dimension has periodic BCs they must be exactly two.
414  std::vector<bool> dim_has_periodic_bcs(dim, 0);
415  for (std::size_t dd = 0; dd < dim; ++dd)
416  if (boundaries_dim[dd].size() != 0)
417  {
418  if (boundaries_dim[dd].size() != 2)
419  throw std::runtime_error{
420  "Extracted "
421  + std::to_string(boundaries_dim[dd].size())
422  + " periodic boundaries "
423  "along cartesian dimension "
424  + std::to_string(dd) };
425 
426  dim_has_periodic_bcs[dd] = 1;
427  }
428 
429  // Periodic conditions are assumed to be checked in order
430  // across Cartesian dimensions.
431  // If there are periodic BCs in all dimensions
432  // starting from 0 up to some dimension, add only those boundaries
433  // If there are gap dimensions between periodic dimensions,
434  // add {-inf, inf} boundaries.
435  bool found_nonperiodic = 0;
436  bool nonconsecutive_dims = 0;
437  for (bool periodic : dim_has_periodic_bcs)
438  {
439  if (found_nonperiodic && periodic)
440  {
441  nonconsecutive_dims = 1;
442  break;
443  }
444  if (!periodic)
445  found_nonperiodic = 1;
446  }
447  std::vector<std::pair<double, double>> boundaries;
448  for (std::size_t dd = 0; dd < dim; ++dd)
449  {
450  if (dim_has_periodic_bcs[dd])
451  {
452  boundaries.push_back({
453  std::min(boundaries_dim[dd][0],
454  boundaries_dim[dd][1]),
455  std::max(boundaries_dim[dd][0],
456  boundaries_dim[dd][1])
457  });
458  }
459  else if (nonconsecutive_dims)
460  {
461  boundaries.push_back({
462  -std::numeric_limits<double>::infinity(),
463  std::numeric_limits<double>::infinity()
464  });
465  }
466  }
467  return { boundaries };
468  }

132  template <typename State, typename TimeGenerator, typename JumpGenerator>
133  double operator()(State &state, TimeGenerator &time_generator,
134  JumpGenerator &jump_generator) const {
135  auto cell_id = state.cell;
136  if constexpr (check_if_outside)
137  if (outside<warn_if_outside>(state.cell, make_point(state.position),
138  "Using nearest cell."))
139  cell_id = _geometry.locator.nearest_cell(state);
140  double cell_side = std::pow(_geometry.mesh().V()[cell_id] / _volume_factor,
141  1. / Geometry::dim);
142  double local_advection_time = cell_side / Foam::mag(_velocity_field(state));
143  double local_diffusion_time = cell_side * cell_side / (2. * _diff_coeff);
144 
145  double reaction_rate = 0.;
146  if constexpr (!std::is_same_v<SurfaceReaction, SurfaceReaction_DoNothing>)
147  for (auto face : _geometry.mesh().cells()[cell_id]) {
148  double rate = _surface_reaction.rate(face);
149  if (rate > reaction_rate)
150  reaction_rate = rate;
151  }
152  double local_reaction_time = _diff_coeff / (reaction_rate * reaction_rate);
153 
154  double time_step =
155  std::max(std::min({_local_time_step_adv * local_advection_time,
156  _local_time_step_diff * local_diffusion_time,
157  _local_time_step_react * local_reaction_time}),
158  _time_step_global);
159 
160  if constexpr (meta::has_time_step_setter_v<TimeGenerator>)
161  time_generator.time_step(time_step);
162  if constexpr (meta::has_time_step_setter_v<JumpGenerator>)
163  jump_generator.time_step(time_step);
164  if constexpr (meta::has_time_step_setter_v<SurfaceReaction>)
165  _surface_reaction.time_step(time_step);
166 
167  return time_step;
168  }

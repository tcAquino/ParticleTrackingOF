159 template <typename Position, typename Sphere>
160 void reflectOffSphere_outsideToInside(Position const &position_new,
161  Position const &position_old,
162  Sphere const &sphere, Position &reflected,
163  Position &contact_point) {
164  // Vector from outside position to inside position
165  auto insideminusoutside = op::minus(position_new, position_old);
166 
167  // Vector from outside position to sphere center
168  auto outsideminuscenter = op::minus(position_old, sphere.center);
169 
170  // Find contact point with sphere
171  double coeff_a = op::abs_sq(insideminusoutside);
172  double coeff_b = 2. * op::dot(insideminusoutside, outsideminuscenter);
173  double coeff_c = op::dot(outsideminuscenter, outsideminuscenter) -
174  sphere.radius * sphere.radius;
175  double fraction_to_intersection =
176  -(coeff_b + std::sqrt(coeff_b * coeff_b - 4. * coeff_a * coeff_c)) /
177  (2. * coeff_a);
178  op::linearop(fraction_to_intersection, insideminusoutside, position_old,
179  contact_point);
180 
181  // Vector from contact point to inside position
182  auto leftover =
183  op::times_scalar(1. - fraction_to_intersection, insideminusoutside);
184 
185  // Compute unit normal to sphere surface at contact
186  auto normal = op::minus(contact_point, sphere.center);
187  op::div_scalar_inplace(normal, sphere.radius);
188 
189  // Compute reflected position
190  op::linearop(-2. * op::dot(leftover, normal), normal, position_new,
191  reflected);
192 }

80 double boundary_reflecting(double position,
81  std::pair<double, double> const &boundaries) {
82  double box_size = boundaries.second - boundaries.first;
83  double pos = position - boundaries.first;
84  double nr_boxes_jumped = std::trunc(pos / box_size);
85  if (int(nr_boxes_jumped) % 2 == 0)
86  return -pos + std::abs(pos - nr_boxes_jumped * box_size);
87  else
88  return -pos + box_size - std::abs(pos - nr_boxes_jumped * box_size);

550 class Boundary_RadialReflecting_2d {
551 public:
552  const double radius;
553 
558  Boundary_RadialReflecting_2d(double radius) : radius{radius} {}
559 
565  template <typename Position>
566  bool out_of_bounds(Position const &position) const {
567  return op::abs_sq(position) > _radius_sq;
568  }
569 
576  template <typename State>
577  bool operator()(State &state, State const &state_old) const {
578  if (!out_of_bounds(state.position))
579  return false;
580 
581  auto position_old = state_old.position;
582  std::vector<double> jump = op::minus(state.position, position_old);
583  std::vector<double> tangent_at_contact(2);
584 
585  // Iterate reflection procedure until position is inside boundary
586  while (1) {
587  // Fraction alpha of jump up to the boundary
588  double pos_dot_jump = op::dot(position_old, jump);
589  double norm_sq_jump = op::abs_sq(jump);
590  double pos_sq_old = op::abs_sq(position_old);
591  double aux = pos_dot_jump / norm_sq_jump;
592  double alpha =
593  std::sqrt(aux * aux + (_radius_sq - pos_sq_old) / norm_sq_jump) - aux;
594 
595  // Avoid numerical issues
596  if (alpha <= 0. || std::isnan(alpha)) {
597  op::times_scalar_inplace(radius / op::abs(state.position),
598  state.position);
599  break;
600  }
601 
602  // Place old position at contact
603  for (std::size_t dd = 0; dd < 2; ++dd)
604  position_old[dd] += alpha * jump[dd];
605 
606  // Tangent to boundary at contact
607  tangent_at_contact[0] = position_old[1] / radius;
608  tangent_at_contact[1] = -position_old[0] / radius;
609 
610  // Angle between jump and tangent to boundary at contact
611  double tangent_dot_jump = op::dot(tangent_at_contact, jump);
612  double cos_contact_angle = tangent_dot_jump / std::sqrt(norm_sq_jump);
613  double contact_angle = std::acos(cos_contact_angle);
614 
615  // Outgoing jump at collision, rotate outer jump and flip the direction
616  op::rotate(jump, -2. * contact_angle);
617  op::times_scalar_inplace(1. - alpha, jump);
618 
619  // Place particle at reflected position
620  op::plus(position_old, jump, state.position);
621 
622  // If inside boundary, done
623  if (!out_of_bounds(state.position))
624  break;
625  }
626 
627  return true;
628  }
629 
630 private:
631  const double _radius_sq{radius * radius};
632 };

223  std::vector<std::vector<int>> projections_backward(dim,
224  std::vector<int>(dim));
225  for (std::size_t dd = 0; dd < dim; ++dd) {
226  projections_forward[dd][dd] = 1.;
227  projections_backward[dd][dd] = -1.;
228  }
229 
230  for (std::size_t pp = 0; pp < nr_points; ++pp) {
231  auto projections =
232  project(points[pp], boundary_periodic.symmetry_planes,
233  boundary_periodic.scale, boundary_periodic.origin);
234  for (std::size_t dd = 0; dd < dim; ++dd) {
235  // If close to lower bound, make an image forward
236  if (projections[dd] < fraction) {
237  points.push_back(points[pp]);

239  boundary_periodic.translate(points.back(), projections_forward[dd]);
240  }
241  // If close to upper bound, make an image backward
242  if (projections[dd] > 1. - fraction) {
243  points.push_back(points[pp]);
244  velocities.push_back(velocities[pp]);
245  boundary_periodic.translate(points.back(), projections_backward[dd]);
246  }

645 template <typename Locator>
646 auto cell_ids_region_cartesian(
647  std::vector<std::pair<double, double>> boundaries, Locator const &locator) {
648  // Identify degenerate dimensions
649  // (if some minimum and maximum boundary values are equal,
650  // the Cartesian region has lower dimension than the space)
651  std::vector<std::size_t> degenerate_dimensions;
652  std::vector<std::size_t> non_degenerate_dimensions;
653  for (std::size_t dd = 0; dd < boundaries.size(); ++dd)
654  if (boundaries[dd].first == boundaries[dd].second)
655  degenerate_dimensions.push_back(dd);
656  else
657  non_degenerate_dimensions.push_back(dd);
658 
659  auto const &mesh = locator.mesh();
660 
661  std::set<Foam::label> cell_ids;
662  for (Foam::label cc = 0; cc < mesh.nCells(); ++cc) {
663  auto center = cell_center(cc, mesh);
664 
665  bool cell_is_within_non_degenerate_boundaries = 1;
666  for (auto dd : non_degenerate_dimensions) {
667  if (center[dd] < boundaries[dd].first ||
668  center[dd] > boundaries[dd].second) {
669  cell_is_within_non_degenerate_boundaries = 0;
670  break;
671  }
672  }
673  if (!cell_is_within_non_degenerate_boundaries)
674  continue;
675  if (degenerate_dimensions.size() == 0)
676  cell_ids.insert(cc);
677 
678  // Consider a position equal to the cell center of the candidate cell
679  // but with components along the degenerate dimension
680  // equal to the prescribed value. If it is in the
681  // mesh, include it in the region
682  for (auto dd : degenerate_dimensions)
683  center[dd] = degenerate_dimensions[dd];
684  auto cell_id = mesh.findCell(center);
685  if (!outside(cell_id))
686  cell_ids.insert(cell_id);
687  }
688 
689  std::vector<Foam::label> cells(cell_ids.begin(), cell_ids.end());
690  return cells;
691 }

840 template <typename Locator>
841 auto cell_ids_region_cartesian(
842  std::vector<std::pair<double, double>> const &boundaries,
843  Locator const &locator) {
844  // Identify degenerate dimensions
845  // (if some minimum and maximum boundary values are equal,
846  // the Cartesian region has lower dimension than the space)
847  std::vector<std::size_t> degenerate_dimensions;
848  std::vector<std::size_t> non_degenerate_dimensions;
849  for (std::size_t dd = 0; dd < boundaries.size(); ++dd)
850  if (boundaries[dd].first == boundaries[dd].second)
851  degenerate_dimensions.push_back(dd);
852  else
853  non_degenerate_dimensions.push_back(dd);
854 
855  auto const &mesh = locator.mesh();
856 
857  std::set<Foam::label> cell_ids;
858  for (Foam::label cc = 0; cc < mesh.nCells(); ++cc) {
859  auto center = cell_center(cc, mesh);
860 
861  bool cell_is_within_non_degenerate_boundaries = 1;
862  for (auto dd : non_degenerate_dimensions) {
863  if (center[dd] < boundaries[dd].first ||
864  center[dd] > boundaries[dd].second) {
865  cell_is_within_non_degenerate_boundaries = 0;
866  break;
867  }
868  }
869  if (!cell_is_within_non_degenerate_boundaries)
870  continue;
871  if (degenerate_dimensions.size() == 0)
872  cell_ids.insert(cc);
873 
874  // Consider a position equal to the cell center of the candidate cell
875  // but with components along the degenerate dimension
876  // equal to the prescribed value. If it is in the
877  // mesh, include it in the region
878  for (auto dd : degenerate_dimensions)
879  center[dd] = degenerate_dimensions[dd];
880  auto cell_id = mesh.findCell(center);
881  if (!outside(cell_id))
882  cell_ids.insert(cell_id);
883  }
884 
885  std::vector<Foam::label> cells(cell_ids.begin(), cell_ids.end());
886  return cells;
887 }

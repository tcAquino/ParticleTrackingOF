547 template <typename Locator>
548 auto cell_ids_region_cartesian(
549  std::vector<std::pair<double, double>> boundaries, Locator const &locator) {
550  // Identify degenerate dimensions
551  // (if some minimum and maximum boundary values are equal,
552  // the Cartesian region has lower dimension than the space)
553  std::vector<std::size_t> degenerate_dimensions;
554  std::vector<std::size_t> non_degenerate_dimensions;
555  for (std::size_t dd = 0; dd < boundaries.size(); ++dd)
556  if (boundaries[dd].first == boundaries[dd].second)
557  degenerate_dimensions.push_back(dd);
558  else
559  non_degenerate_dimensions.push_back(dd);
560 
561  auto const &mesh = locator.mesh();
562 
563  std::set<Foam::label> cell_ids;
564  for (Foam::label cc = 0; cc < mesh.nCells(); ++cc) {
565  auto center = cell_center(cc, mesh);
566 
567  bool cell_is_within_non_degenerate_boundaries = 1;
568  for (auto dd : non_degenerate_dimensions) {
569  if (center[dd] < boundaries[dd].first ||
570  center[dd] > boundaries[dd].second) {
571  cell_is_within_non_degenerate_boundaries = 0;
572  break;
573  }
574  }
575  if (!cell_is_within_non_degenerate_boundaries)
576  continue;
577  if (degenerate_dimensions.size() == 0)
578  cell_ids.insert(cc);
579 
580  // Consider a position equal to the cell center of the candidate cell
581  // but with components along the degenerate dimension
582  // equal to the prescribed value. If it is in the
583  // mesh, include it in the region
584  for (auto dd : degenerate_dimensions)
585  center[dd] = degenerate_dimensions[dd];
586  auto cell_id = mesh.findCell(center);
587  if (!outside(cell_id))
588  cell_ids.insert(cell_id);
589  }
590 
591  std::vector<Foam::label> cells(cell_ids.begin(), cell_ids.end());
592  return cells;
593 }

41 template <typename Transport, typename Solvers, typename VelocityField,
42  typename Geometry, typename Boundary, typename ReactionParameters,
43  typename BulkReaction>
44 auto makeTransitions(VelocityField const &velocity_field,
45  Geometry const &geometry, Boundary &boundary,
46  typename Transport::Parameters const &params_transport,
47  ReactionParameters const &params_reaction,
48  typename Solvers::Parameters const &params_solvers,
49  BulkReaction const &bulk_reaction) {
50  using Transitions = decltype(ctrw::Transitions_CTRW_Transport_Reaction{
51  Transport::template makeTransitions<Solvers>(
52  velocity_field, geometry, boundary, params_transport, params_reaction,
53  params_solvers),
54  bulk_reaction});
55  std::vector<Transitions> transitions;
56  for (std::size_t thread = 0;
57  thread < useful::get_num_threads(meta::ParallelOptions::Parallel{});
58  ++thread)
59  transitions.emplace_back(Transport::template makeTransitions<Solvers>(
60  velocity_field, geometry, boundary,
61  params_transport, params_reaction,
62  params_solvers),
63  bulk_reaction);
64 
65  return transitions;
66 }

426 std::vector<std::pair<double, double>> pdf(std::vector<double> const &bin_edges,
427  Container const &data) {
428  std::vector<std::pair<double, double>> pdf(bin_edges.size() - 1);
429 
430  // Populate all bins except last
431  auto rt_it = data.cbegin();
432  for (std::size_t bin = 0; bin < bin_edges.size() - 2; ++bin) {
433  pdf[bin].first = (bin_edges[bin] + bin_edges[bin + 1]) / 2.;
434  while (rt_it != data.cend() && *rt_it < bin_edges[bin + 1]) {
435  ++pdf[bin].second;
436  ++rt_it;
437  }
438  pdf[bin].second /= bin_edges[bin + 1] - bin_edges[bin];
439  }
440 
441  // Put everything that is left on last bin
442  for (; rt_it != data.cend(); ++rt_it) {
443  pdf.back().first =
444  (bin_edges[bin_edges.size() - 2] + bin_edges.back()) / 2.;
445  ++pdf.back().second;
446  }
447  pdf.back().second /= bin_edges.back() - bin_edges[bin_edges.size() - 2];

507  std::size_t nr_bins) {
508  std::pair<std::vector<double>, std::vector<double>> cdf;
509  cdf.first.reserve(nr_bins);
510  cdf.second.reserve(nr_bins);
511  std::sort(samples);
512 
513  std::vector<double> edges;
514  if constexpr (std::is_same<Spacing, Logspacing>::value)
515  edges = range::logspace(min_edge, max_edge, nr_bins + 1);
516  else if constexpr (std::is_same<Spacing, Logspacing>::value)
517  edges = range::linspace(min_edge, max_edge, nr_bins + 1);
518  else
519  throw useful::bad_parameters();
520 
521  for (std::size_t ii = 1; ii < edges.size(); ++ii)
522  cdf.first.push_back(edges[ii] + edges[ii - 1] / 2.);
523 
524  auto start =
525  std::lower_bound(std::begin(samples), std::end(samples), edges[0]);
526  for (std::size_t ii = 1; ii < edges.size(); ++ii) {
527  double freq = double(std::lower_bound(std::begin(samples),
528  std::end(samples), edges[ii]) -
529  start) /
530  samples.size();
531  cdf.second.push_back(freq);
532  }
533 
534  return cdf;

542  double max_factor = 1.1) {

548 
549  std::size_t nr_samples = samples.size();
550  std::size_t zero_samples = 0;
551  if constexpr (std::is_same<Spacing, Logspacing>::value) {
552  samples.erase(std::remove(std::begin(samples), std::end(samples), 0.),
553  std::end(samples));
554  zero_samples = nr_samples - samples.size();
555  }
556 
557  double min_edge = min_factor * samples[0];
558  double max_edge = max_factor * samples[samples.size() - 1];
559 
560  std::vector<double> edges;
561  if constexpr (std::is_same<Spacing, Logspacing>::value)
562  edges = range::logspace(min_edge, max_edge, nr_bins + 1);

566  throw useful::bad_parameters();

569  cdf.first.push_back((edges[ii] + edges[ii - 1]) / 2.);

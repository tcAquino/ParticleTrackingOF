538 template <typename Container>
539 std::vector<std::pair<double, double>> pdf(std::vector<double> const &bin_edges,
540  Container const &data) {
541  std::vector<std::pair<double, double>> pdf(bin_edges.size() - 1);
542 
543  // Populate all bins except last
544  auto rt_it = data.cbegin();
545  for (std::size_t bin = 0; bin < bin_edges.size() - 2; ++bin) {
546  pdf[bin].first = (bin_edges[bin] + bin_edges[bin + 1]) / 2.;
547  while (rt_it != data.cend() && *rt_it < bin_edges[bin + 1]) {
548  ++pdf[bin].second;
549  ++rt_it;
550  }
551  pdf[bin].second /= bin_edges[bin + 1] - bin_edges[bin];
552  }
553 
554  // Put everything that is left on last bin
555  for (; rt_it != data.cend(); ++rt_it) {
556  pdf.back().first =
557  (bin_edges[bin_edges.size() - 2] + bin_edges.back()) / 2.;
558  ++pdf.back().second;
559  }
560  pdf.back().second /= bin_edges.back() - bin_edges[bin_edges.size() - 2];
561 
562  return pdf;
563 }

668  double max_factor = 1.1) {
669  std::pair<std::vector<double>, std::vector<double>> cdf;
670

672  cdf.second.reserve(nr_bins);
673  std::sort(std::begin(samples), std::end(samples));
674 
675  std::size_t nr_samples = samples.size();
676  std::size_t zero_samples = 0;
677  if constexpr (std::is_same<Spacing, Logspacing>::value) {
678  samples.erase(std::remove(std::begin(samples), std::end(samples), 0.),
679  std::end(samples));
680  zero_samples = nr_samples - samples.size();
681  }
682 
683  double min_edge = min_factor * samples[0];
684  double max_edge = max_factor * samples[samples.size() - 1];
685 
686  std::vector<double> edges;
687  if constexpr (std::is_same<Spacing, Logspacing>::value)
688  edges = range::logspace(min_edge, max_edge, nr_bins + 1);
689  else if constexpr (std::is_same<Spacing, Linspacing>::value)
690  edges = range::linspace(min_edge, max_edge, nr_bins + 1);
691  else
692  throw io::bad_parameters();
693 
694  for (std::size_t ii = 1; ii < edges.size(); ++ii)
695  cdf.first.push_back((edges[ii] + edges[ii - 1]) / 2.);
696 
697  for (std::size_t ii = 1; ii < edges.size(); ++ii) {
698  double freq = double(std::lower_bound(std::begin(samples),
699  std::end(samples), edges[ii]) -
700  std::begin(samples)) /
701  nr_samples;
702  cdf.second.push_back(freq);
703  }

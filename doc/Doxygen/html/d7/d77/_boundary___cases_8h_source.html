85  template <typename State>
86  bool operator()(State &state, State const &state_old = {}) {
87  // Find first intersection with boundary patch.
88  auto intersection = _locator.mesh_search().intersection(
89  make_point(state_old.position), make_point(state.position));
90 
91  // When the start point is on a cell face,
92  // sometimes the intersection with it is not found, and sometimes it is
93  if (!intersection.hit()) {
94  // Note: The code flow is chosen to avoid trying to locate particles which
95  // are outside as much as possible, because it is expensive
96  state.cell = _locator(state);
97  if (outside(state.cell)) {
98  // If the final state is outside, avoid particles leaving the domain by
99  // checking for intersections with a small backwards offset
100  intersection = _locator.mesh_search().intersection(
101  offset_backward_cell(make_point(state_old.position), state_old.cell,
102  make_point(state.position) -
103  make_point(state_old.position),
104  _locator),
105  make_point(state.position));
106  // If this didn't work, place particle at nearest cell center to avoid
107  // breaking
108  if (!intersection.hit())
109  state.set_position(cell_center(_locator.nearest_cell(state.position),
110  _locator.mesh()));
111  }
112  } else if (((make_point(state.position) - intersection.point()) &
113  unit_normal_outward(intersection.index(), _locator.mesh())) <
114  0.) {
115  // If the displacement is against the local face normal (into the domain
116  // at the boundary), ignore intersection and find
117  // the next intersection with a boundary patch
118  auto old_intersection = intersection;
119  intersection =
120  next_intersection(intersection, make_point(state.position));
121 
122  // Ignore new intersection if offset went beyond final point or if there
123  // is not enough precision to find a different intersection
124  if (next_intersection_is_beyond_final_point(
125  intersection, old_intersection.point(), state) ||
126  intersection.index() == old_intersection.index())
127  intersection.setMiss();
128 
129  // If there was no new hit and the particle is still outside, there is a
130  // precision issue in finding intersections for the mesh. Place particle
131  // at nearest cell center.
132  if (!intersection.hit() && outside(_locator(state))) {
133  state.cell = _locator.nearest_cell(state.position);
134  state.set_position(cell_center(state.cell, _locator.mesh()));
135  return true;
136  }
137  }
138 
139  bool had_effect = 0;
140 
141  // While some patch is intersected
142  while (intersection.hit()) {
144  auto patch_id =
145  _locator.mesh().boundaryMesh().whichPatch(intersection.index());
146  auto patch = patch_name(patch_id);
147  auto type_name = _boundary_conditions.at(patch);
148 
149  // Apply the approriate boundary and store associated info
150  switch (_boundary_condition_types.type(type_name)) {
151  case BoundaryConditionList::Type::reflecting: {
152  _store_info(state, state_old, intersection, _boundary_condition_types,
153  meta::Selector<BoundaryConditionList::Type,
154  BoundaryConditionList::Type::reflecting>{});
155  boundary_reflecting(state, intersection.point(),
156  reflection_normal(intersection.index()));
157  had_effect = 1;
158  break;
159  }
160  case BoundaryConditionList::Type::reacting_reflecting: {
161  _store_info(state, state_old, intersection, _boundary_condition_types,
162  meta::Selector<BoundaryConditionList::Type,
163  BoundaryConditionList::Type::reflecting>{});
164  surface_reaction(state, state_old, intersection.index());
165  boundary_reflecting(state, intersection.point(),
166  reflection_normal(intersection.index()));
167  had_effect = 1;
168  break;
169  }
170  case BoundaryConditionList::Type::periodic: {
171  _store_info(state, state_old, intersection, _boundary_condition_types,
172  meta::Selector<BoundaryConditionList::Type,
173  BoundaryConditionList::Type::periodic>{});
174  had_effect += _boundary_periodic(state, intersection);
175  break;
176  }
177  case BoundaryConditionList::Type::absorbing: {
178  _store_info(state, state_old, intersection, _boundary_condition_types,
179  meta::Selector<BoundaryConditionList::Type,
180  BoundaryConditionList::Type::absorbing>{});
181  state.set_position(intersection.point());
182  had_effect = 1;
183  break;
184  }
185  case BoundaryConditionList::Type::custom: {
186  _store_info(state, state_old, intersection, _boundary_condition_types,
187  meta::Selector<BoundaryConditionList::Type,
188  BoundaryConditionList::Type::custom>{});
189  had_effect += _boundary_custom(state, state_old, intersection);
190  break;
191  }
192  case BoundaryConditionList::Type::empty: {
193  }
194  default:
195  throw std::runtime_error{"Boundary condition type " + type_name +
196  " not supported"};
197  }
198 
199  // Stop if right at the last intersection point
200  // This avoids numerical issues for some edge cases
201  if (make_point(state.position) == intersection.point())
202  break;
203 
204  // Find the next intersection with a boundary patch
205  auto old_intersection = intersection;
206  intersection =
207  next_intersection(intersection, make_point(state.position));
208  // Ignore new intersection if offset went beyond final point or if there
209  // is not enough precision to find a different intersection
210  if (next_intersection_is_beyond_final_point(
211  intersection, old_intersection.point(), state) ||
212  intersection.index() == old_intersection.index())
213  intersection.setMiss();
214 
215  // Sometimes there is not enough precision to find the next intersection.
216  // Avoid particles leaving the domain by placing the final state at the
217  // last intersection
218  if (!intersection.hit()) {
219  state.cell = _locator(state);
220  if (outside(state.cell))
221  state.set_position(old_intersection.point());
222  break;
223  }
224  }
225 
226  return had_effect;
227  }

370  template <typename IStream, typename TransportParameters,
371  typename ReactionParameters>
372  void read_measurement_spacing(IStream &input, std::string const &filename,
373  TransportParameters const &params_transport,
374  ReactionParameters const &params_reaction) {
375  std::string in_file = std::string{"In file "} + filename + " : ";
376  auto split_line = io::split_line(input);
377  std::size_t param_index = 0;
378  io::read(split_line, param_index,
379  in_file + "Could not parse measurement spacing",
380  measurement_spacing);
381  std::string for_measurement_spacing =
382  std::string{"Measurement spacing "} + measurement_spacing + " : ";
383 
384  io::read(split_line, param_index,
385  in_file + for_measurement_spacing +
386  "Could not parse minimum measurement time",
387  time_min);
388  time_min *= time_unit_factor;
389  switch (MeasurementSpacing::type(measurement_spacing)) {
390  case MeasurementSpacing::Type::linear_step: {
391  io::read(split_line, param_index, in_file + "Could not parse time step",
392  time_increment);
393  time_increment *= time_unit_factor;
394  time_max = std::numeric_limits<double>::infinity();
395  if (!(time_increment > 0.))
396  throw std::runtime_error{in_file + for_measurement_spacing +
397  "Non-positive time increment"};
398  break;
399  }
400  case MeasurementSpacing::Type::log_step: {
401  io::read(split_line, param_index,
402  in_file + for_measurement_spacing +
403  "Could not parse time step factor",
404  time_increment);
405  time_max = std::numeric_limits<double>::infinity();
406  if (!(time_increment > 1.))
407  throw std::runtime_error{in_file + for_measurement_spacing +
408  "Time increment factor not greater than one"};
409  break;
410  }
411  case MeasurementSpacing::Type::linear: {
412  std::size_t nr_measurements;
413  io::read(split_line, param_index,
414  in_file + for_measurement_spacing +
415  "Could not parse maximum time and number of measurements",
416  time_max, nr_measurements);
417  time_max *= time_unit_factor;
418  if (nr_measurements < 2)
419  throw std::runtime_error{in_file + for_measurement_spacing +
420  "Number of measurements not greater than one"};
421  time_increment = (time_max - time_min) / (nr_measurements - 1);
422  if (!(time_increment > 0.))
423  throw std::runtime_error{in_file + for_measurement_spacing +
424  "Non-positive time increment"};
425  break;
426  }
427  case MeasurementSpacing::Type::log: {
428  if (!(time_min > 0.))
429  throw std::runtime_error{in_file + for_measurement_spacing +
430  "Non-positive minimum measurement time"};
431  std::size_t nr_measurements;
432  io::read(split_line, param_index,
433  in_file + for_measurement_spacing +
434  "Could not parse maximum time and number of measurements",
435  time_max, nr_measurements);
436  time_max *= time_unit_factor;
437  if (nr_measurements < 2)
438  throw std::runtime_error{in_file + for_measurement_spacing +
439  "Number of measurements not greater than one"};
440  time_increment =
441  std::pow(time_max / time_min, 1. / (nr_measurements - 1));
442  if (!(time_increment > 1.))
443  throw std::runtime_error{in_file + for_measurement_spacing +
444  "Time increment factor not greater than one"};
445  break;
446  }
447  default:
448  throw std::runtime_error{in_file + for_measurement_spacing +
449  "Not supported"};
450  }

454  template <typename IStream, typename Geometry>
455  void read_measurement_types(IStream &input, std::string const &filename,
456  Geometry const &geometry) {
457  std::string in_file = std::string{"In file "} + filename + " : ";
458  for (std::vector<std::string> split_line; io::split_line(input, split_line);
459  split_line.clear()) {
460  std::size_t param_index = 0;
461  auto name =
462  io::read<std::string>(split_line, param_index,
463  in_file + "Could not parse measurement type");
464  std::string for_measurement_type =
465  std::string{"Measurement type "} + name + " : ";
466  if (!MeasurementList{}.contains(name))
467  throw std::runtime_error{"Output type " + name + " not supported"};
468 
469  switch (MeasurementList::type(name)) {
470  case (MeasurementList::Type::scalar_field):
471  case (MeasurementList::Type::vector_field):
472  case (MeasurementList::Type::tensor_field): {
473  auto field_name = io::read<std::string>(
474  split_line, param_index,
475  in_file + for_measurement_type + "Could not parse field name");
476  measurements.emplace_back(
477  std::make_unique<Measurement_Field>(name, field_name));
478  break;
479  }
480  case (MeasurementList::Type::first_crossing_time): {
481  auto dim = io::read<double>(split_line, param_index,
482  in_file + for_measurement_type +
483  "Could not parse crossing dimension");
484  auto position =
485  io::read<double>(split_line, param_index,
486  in_file + for_measurement_type +
487  "Could not parse crossing position");
488  if (dim >= Geometry::dim)
489  throw std::runtime_error{
490  in_file + for_measurement_type +
491  "Crossing dimension higher than simulation dimension"};
492  measurements.emplace_back(
493  std::make_unique<Measurement_Dim_Position>(name, dim, position));
494  break;
495  }
496  case (MeasurementList::Type::position_nth_moment):
497  case (MeasurementList::Type::position_nth_moment_periodic): {
498  auto order = io::read<double>(split_line, param_index,
499  in_file + for_measurement_type +
500  "Could not parse moment order");
501  measurements.emplace_back(
502  std::make_unique<Measurement_Order>(name, order));
503  break;
504  }
505  case (MeasurementList::Type::position_moment):
506  case (MeasurementList::Type::position_moment_periodic): {
507  auto orders = io::read<double>(split_line, param_index, Geometry::dim,
508  in_file + for_measurement_type +
509  "Could not parse moment coordinate "
510  "orders along each dimension");
511  measurements.emplace_back(
512  std::make_unique<Measurement_Orders>(name, orders));
513  break;
514  }
515  default:
516  measurements.emplace_back(std::make_unique<Measurement>(name));
517  }
518 
519  if (param_index < split_line.size())
520  io::read(split_line, param_index,
521  in_file + for_measurement_type + "Could not parse precision",
522  measurements.back()->precision);
523  }
524  }

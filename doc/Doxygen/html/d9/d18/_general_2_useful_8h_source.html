33 inline std::ostream &display_duration(std::ostream &stream,
34  std::chrono::nanoseconds ns) {
35  io::StreamScopeFormat guard{stream};
36  using days = std::chrono::duration<std::int_least64_t, std::ratio<86400>>;
37  stream.fill('0');
38  stream << std::right;
39  auto dd = std::chrono::duration_cast<days>(ns);
40  ns -= dd;
41  auto hh = std::chrono::duration_cast<std::chrono::hours>(ns);
42  ns -= hh;
43  auto mm = std::chrono::duration_cast<std::chrono::minutes>(ns);
44  ns -= mm;
45  auto ss = std::chrono::duration_cast<std::chrono::seconds>(ns);
46  ns -= ss;
47  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(ns);
48  stream << std::setw(2) << dd.count() << "d:" << std::setw(2) << hh.count()
49  << "h:" << std::setw(2) << mm.count() << "m:" << std::setw(2)
50  << ss.count() << "s:" << std::setw(3) << ms.count() << "ms";
51 
52  return stream;
53 };

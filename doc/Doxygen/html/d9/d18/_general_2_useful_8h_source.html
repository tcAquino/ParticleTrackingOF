35 inline std::ostream &display_duration(std::ostream &stream,

37  using days = std::chrono::duration<int, std::ratio<86400>>;
38  char fill = stream.fill();
39  stream.fill('0');
40  auto d = std::chrono::duration_cast<days>(ns);
41  ns -= d;
42  auto h = std::chrono::duration_cast<std::chrono::hours>(ns);
43  ns -= h;
44  auto m = std::chrono::duration_cast<std::chrono::minutes>(ns);
45  ns -= m;
46  auto s = std::chrono::duration_cast<std::chrono::seconds>(ns);
47  ns -= s;
48  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(ns);
49  stream << std::setw(2) << d.count() << "d:" << std::setw(2) << h.count()
50  << "h:" << std::setw(2) << m.count() << "m:" << std::setw(2)
51  << s.count() << "s:" << std::setw(3) << ms.count() << "ms";
52  stream.fill(fill);
53

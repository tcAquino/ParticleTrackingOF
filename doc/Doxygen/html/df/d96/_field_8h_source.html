509 template <typename Field, typename Mesh>
510 auto magnitude_of_average(Field &field, Mesh const &mesh) {
511  Foam::scalar mesh_volume = Foam::sum(mesh.cellVolumes());
512  auto average_weighted_data = Foam::sum(field * mesh.cellVolumes());
513  return Foam::mag(average_weighted_data) / mesh_volume;
514 }

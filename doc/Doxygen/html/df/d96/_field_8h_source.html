642 template <typename Field, typename Mesh>
643 auto magnitude_of_average(Field &field, Mesh const &mesh) {
644  Foam::scalar mesh_volume = Foam::sum(mesh.cellVolumes());
645  auto average_weighted_data = Foam::sum(field * mesh.cellVolumes());
646  return Foam::mag(average_weighted_data) / mesh_volume;
647 }
648 
651 template <typename Field, typename Mesh>
652 void rescale_to_average(Field &field, Mesh const &mesh, double average) {
653  field *= average / magnitude_of_average(field, mesh);
654 }

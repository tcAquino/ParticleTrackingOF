255  template <typename IStream, typename TransportParameters,
256  typename ReactionParameters>
257  void read_measure_spacing(IStream &input,
258  TransportParameters const &params_transport,
259  ReactionParameters const &params_reaction) {
260  useful::read_first_from_line(input, measure_spacing, comment_sequence);
261  useful::read_first_from_line(input, time_min, comment_sequence);
262  time_min *= time_unit_factor;
263  switch (MeasureSpacing::type(measure_spacing)) {
264  case MeasureSpacing::Type::step: {
265  useful::read_first_from_line(input, time_increment, comment_sequence);
266  time_increment *= time_unit_factor;
267  time_max = std::numeric_limits<double>::infinity();
268  if (!(time_increment > 0.))
269  throw std::runtime_error{
270  std::string("Non-positive time increment for ") + measure_spacing +
271  " measurement spacing"};
272  break;
273  }
274  case MeasureSpacing::Type::linear: {
275  useful::read_first_from_line(input, time_max, comment_sequence);
276  time_max *= time_unit_factor;
277  std::size_t nr_measures;
278  useful::read_first_from_line(input, nr_measures, comment_sequence);
279  if (nr_measures < 2)
280  throw std::runtime_error{
281  std::string("Minimum of two measurements required for ") +
282  measure_spacing + " measurement spacing"};
283  time_increment = (time_max - time_min) / (nr_measures - 1);
284  if (!(time_increment > 0.))
285  throw std::runtime_error{
286  std::string("Non-positive time increment for ") + measure_spacing +
287  " measurement spacing"};
288  break;
289  }
290  case MeasureSpacing::Type::log: {
291  if (!(time_min > 0.))
292  throw std::runtime_error{
293  std::string("Non-positive minimum measurement time for ") +
294  measure_spacing + " measurement spacing"};
295  useful::read_first_from_line(input, time_max, comment_sequence);
296  time_max *= time_unit_factor;
297  std::size_t nr_measures;
298  useful::read_first_from_line(input, nr_measures, comment_sequence);
299  if (nr_measures < 2)
300  throw std::runtime_error{
301  std::string("Minimum of two measurements required for ") +
302  measure_spacing + " measurement spacing"};
303  time_increment = std::pow(time_max / time_min, 1. / (nr_measures - 1));
304  if (!(time_increment > 1.))
305  throw std::runtime_error{
306  std::string("Non-positive time increment for ") + measure_spacing +
307  " measurement spacing"};
308  break;
309  }
310  default:
311  throw std::runtime_error{std::string("Measure spacing ") +
312  measure_spacing + " not supported"};
313  }
314  }

317  template <typename IStream> void read_measurement_types(IStream &input) {
318  std::string line;
319  while (std::getline(input, line)) {
320  if (line.find(comment_sequence) == 0)
321  continue;
322  useful::remove_carriage_return_in_place(line);
323  useful::clear_escape_in_place(line, comment_sequence);
324  auto split_line = useful::split(line);
325  std::size_t required_size = 1;
326  if (split_line.size() < required_size)
327  throw std::runtime_error{"Could not parse measurement type"};
328  std::string const &name = split_line[0];
329  measurements.emplace_back(name);
330  if (name == "scalar_field" || name == "vector_field" ||
331  name == "tensor_field") {
332  required_size = 2;
333  if (split_line.size() < required_size)
334  throw std::runtime_error{std::string("Measurement type ") + name +
335  ": "
336  "Could not parse field name"};
337  measurements.back().field_name = split_line[1];
338  } else if (name == "first_crossing_time") {
339  required_size = 3;
340  if (split_line.size() < required_size)
341  throw std::runtime_error{
342  std::string("Measurement type ") + name +
343  ": "
344  "Could not parse crossing dimension and/or position"};
345  measurements.back().uint_params.push_back(std::stoul(split_line[1]));
346  measurements.back().double_params.push_back(std::stod(split_line[2]));
347  } else if (name == "position_nth_moment" ||
348  name == "position_nth_moment_periodic") {
349  required_size = 2;
350  if (split_line.size() < required_size)
351  throw std::runtime_error{std::string("Measurement type ") + name +
352  ": "
353  "Could not parse moment order"};
354  measurements.back().double_params.push_back(std::stod(split_line[1]));
355  } else if (name == "position_moment" ||
356  name == "position_moment_periodic") {
357  required_size = 2;
358  if (split_line.size() < required_size)
359  throw std::runtime_error{std::string("Measurement type ") + name +
360  ": "
361  "Could not parse coordinate moment orders"};
362  for (std::size_t ii = 1; ii < split_line.size(); ++ii)
363  measurements.back().double_params.push_back(
364  std::stod(split_line[ii]));
365  }
366  if (split_line.size() > required_size)
367  measurements.back().precision = std::stod(split_line.back());
368  }
369  }

622  std::vector<std::reference_wrapper<const Mask>> masks = {},
623  std::vector<double> thresholds = {}) {
624  for (auto const &measurement : parameters.measurements) {
625  switch (Measure::type(measurement.name)) {
626  case Measure::Type::position: {
627  _output_time.emplace_back(

652  break;
653  }
654  case Measure::Type::position_second_moment: {
655  _output_time.emplace_back(
656  std::make_unique<
657  MeasurerTime_position_second_moment<Subject, Geometry>>(
658  subject, geometry, directories, identifier,
659  measurement.precision));
660  break;
661  }
662  case Measure::Type::position_nth_moment: {
663  if (measurement.double_params.size() == 0)
664  throw std::runtime_error{std::string("Measurement type ") +
665  measurement.name +
666  ": "
667  "Moment order not provided"};
668  _output_time.emplace_back(
669  std::make_unique<

673  break;
674  }

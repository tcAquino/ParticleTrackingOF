142  if (cell_ids.size() == 0 && nr_particles != 0)

784  tag_data.push_back(std::stod(split_line.back()));

841  region_boundaries.push_back(

843  }

854  useful::read_first_from_line(input, time_min, comment_sequence);

856  type == InitialConditions::Type::fluxweighted_inlet_continuous) {

866  time_step_accuracy_diff * params_transport.diffusion_time,

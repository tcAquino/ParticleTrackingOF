46  template <typename Geometry, typename TransportParameters,
47  typename ReactionParameters>
48  InitialConditionParameters_Cases(Directories const &directories,
49  std::string const &name,
50  Geometry const &geometry,
51  TransportParameters const &params_transport,
52  ReactionParameters const &params_reaction) {
53  std::string ic_name;
54  auto input =
55  useful::open_read(directories.dir_parameters +
56  "/parameters_initial_condition_" + name + ".dat");
57  useful::read_first_from_line(ic_name, input);
58  if (!InitialConditionList{}.contains(ic_name))
59  throw std::runtime_error{"Initial condition type " + ic_name +
60  " not supported"};
61  type = InitialConditionList::type(ic_name);
62  if (type == InitialConditionList::Type::uniform_near_solid)
63  useful::read_first_from_line(distance_wall, input);
64  if (type == InitialConditionList::Type::uniform_region_cartesian ||
65  type == InitialConditionList::Type::fluxweighted_region_cartesian) {
66  auto split_line = useful::split_line(input);
67  if (split_line.size() % 2 != 0)
68  std::runtime_error{
69  "Initial condition region boundaries must come in pairs"};
70  for (std::size_t ii = 0; ii < split_line.size(); ii += 2) {
71  region_boundaries.push_back(
72  {std::stod(split_line[ii]), std::stod(split_line[ii + 1])});
73  }
74  }
75  if (type == InitialConditionList::Type::prescribed_positions) {
76  useful::read_first_from_line(filename_data, input);
77  useful::expand_env_in_place(
78  useful::expand_home_dir_in_place(filename_data));
79  }
80  if (type != InitialConditionList::Type::prescribed_positions_masses &&
81  type == InitialConditionList::Type::prescribed_positions_masses_tags)
82  useful::read_first_from_line(initial_mass, input);
83  useful::read_first_from_line(time_min, input);
84  if (type == InitialConditionList::Type::uniform_inlet_continuous ||
85  type == InitialConditionList::Type::fluxweighted_inlet_continuous) {
86  useful::read_first_from_line(time_max, input);
87  useful::read_first_from_line(time_step_accuracy_adv, input);
88  useful::read_first_from_line(time_step_accuracy_diff, input);
89  useful::read_first_from_line(time_step_accuracy_react, input);
90  time_step =
91  std::min({time_step_accuracy_adv * params_transport.advection_time,
92  time_step_accuracy_diff * params_transport.diffusion_time,
93  time_step_accuracy_react * params_reaction.reaction_time});
94  }
95  }

296  template <typename ParticleMakerOther>
297  auto make_particles(std::size_t nr_particles, InitialConditionList::Type type,
298  ParticleMakerOther &particle_maker) {
299  switch (type) {
300  case InitialConditionList::Type::uniform:
301  if constexpr (std::is_same_v<Mask, useful::Empty>)
302  return uniform(nr_particles, _locator.mesh(), _rng, particle_maker);
303  else
304  return uniform(nr_particles, _locator.mesh(), _rng, particle_maker,
305  _mask, _threshold);
306  case InitialConditionList::Type::fluxweighted:
307  if constexpr (std::is_same_v<Mask, useful::Empty>)
308  return fluxweighted(nr_particles, _velocity_field, _locator.mesh(),
309  _rng, particle_maker);
310  else
311  return fluxweighted(nr_particles, _velocity_field, _locator.mesh(),
312  _rng, particle_maker, _mask, _threshold);
313  case InitialConditionList::Type::uniform_inlet:
314  if constexpr (std::is_same_v<Mask, useful::Empty>)
315  return uniform_patches(nr_particles, {"inlet"}, _locator, _rng,
316  particle_maker);
317  else
318  return uniform_patches(nr_particles, {"inlet"}, _locator, _rng,
319  particle_maker, _mask, _threshold);
320  case InitialConditionList::Type::fluxweighted_inlet:
321  if constexpr (std::is_same_v<Mask, useful::Empty>)
322  return fluxweighted_patches(nr_particles, {"inlet"}, _velocity_field,
323  _locator, _rng, particle_maker);
324  else
325  return fluxweighted_patches(nr_particles, {"inlet"}, _velocity_field,
326  _locator, _rng, particle_maker, _mask,
327  _threshold);
328  case InitialConditionList::Type::uniform_solid:
329  if constexpr (std::is_same_v<Mask, useful::Empty>)
330  return uniform_patches(nr_particles, {"wallFluidSolid"}, _locator, _rng,
331  particle_maker);
332  else
333  return uniform_patches(nr_particles, {"wallFluidSolid"}, _locator, _rng,
334  particle_maker, _mask, _threshold);
335  case InitialConditionList::Type::uniform_near_solid:
336  if constexpr (std::is_same_v<Mask, useful::Empty>)
337  return uniform_near_boundary_patches(nr_particles, {"wallFluidSolid"},
338  parameters.distance_wall, _locator,
339  _rng, particle_maker);
340  else
341  return uniform_near_boundary_patches(
342  nr_particles, {"wallFluidSolid"}, parameters.distance_wall,
343  _locator, _rng, particle_maker, _mask, _threshold);
344  case InitialConditionList::Type::uniform_region_cartesian:
345  if constexpr (std::is_same_v<Mask, useful::Empty>)
346  return uniform_cells(
347  nr_particles,
348  cell_ids_region_cartesian(parameters.region_boundaries, _locator),
349  _locator.mesh(), _rng, particle_maker);
350  else
351  return uniform_cells(
352  nr_particles,
353  apply_mask(cell_ids_region_cartesian(parameters.region_boundaries,
354  _locator),
355  _mask, _threshold),
356  _locator.mesh(), _rng, particle_maker);
357  case InitialConditionList::Type::fluxweighted_region_cartesian:
358  if constexpr (std::is_same_v<Mask, useful::Empty>)
359  return fluxweighted_cells(
360  nr_particles,
361  cell_ids_region_cartesian(parameters.region_boundaries, _locator),
362  _velocity_field, _locator.mesh(), _rng, particle_maker);
363  else
364  return fluxweighted_cells(
365  nr_particles,
366  apply_mask(cell_ids_region_cartesian(parameters.region_boundaries,
367  _locator),
368  _mask, _threshold),
369  _velocity_field, _locator.mesh(), _rng, particle_maker);
370  case InitialConditionList::Type::prescribed_positions:
371  return prescribed_positions(nr_particles, parameters.filename_data,
372  particle_maker);
373  case InitialConditionList::Type::prescribed_positions_masses:
374  return prescribed_positions_masses(nr_particles, parameters.filename_data,
375  particle_maker);
376  case InitialConditionList::Type::prescribed_positions_masses_tags:
377  return prescribed_positions_masses_tags(
378  nr_particles, parameters.filename_data, particle_maker);
379  case InitialConditionList::Type::uniform_inlet_continuous:
380  if constexpr (std::is_same_v<Mask, useful::Empty>)
381  return continuous_injection(
382  [this](std::size_t nr_particles,
383  ParticleMakerOther &particle_maker) {
384  return uniform_patches(nr_particles, {"inlet"}, _locator, _rng,
385  particle_maker);
386  },
387  nr_particles, parameters.time_min, parameters.time_max,
388  parameters.time_step, particle_maker);
389  else
390  return continuous_injection(
391  [this](std::size_t nr_particles,
392  ParticleMakerOther &particle_maker) {
393  return uniform_patches(nr_particles, {"inlet"}, _locator, _rng,
394  particle_maker, _mask, _threshold);
395  },
396  nr_particles, parameters.time_min, parameters.time_max,
397  parameters.time_step, particle_maker);
398  case InitialConditionList::Type::fluxweighted_inlet_continuous:
399  if constexpr (std::is_same_v<Mask, useful::Empty>)
400  return continuous_injection(
401  [this](std::size_t nr_particles,
402  ParticleMakerOther &particle_maker) {
403  return fluxweighted_patches(nr_particles, {"inlet"},
404  _velocity_field, _locator, _rng,
405  particle_maker);
406  },
407  nr_particles, parameters.time_min, parameters.time_max,
408  parameters.time_step, particle_maker);

412  ParticleMakerOther &particle_maker) {
413  return fluxweighted_patches(nr_particles, {"inlet"},
414  _velocity_field, _locator, _rng,
415  particle_maker, _mask, _threshold);

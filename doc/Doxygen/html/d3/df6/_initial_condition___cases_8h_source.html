544  TransportParameters const &params_transport,
545  ReactionParameters const &params_reaction) {

547  InitialConditionList::Type::prescribed_positions_masses_tags_times) {
548  continuity = "prescribed";
549  return std::make_unique<InjectionParameters>();
550  }
551

553  auto split_line = io::split_line(input);
554  std::size_t param_index = 0;
555  io::read(split_line, param_index,
556  std::string{"In file "} + filename +
557  " : "
558  "Could not parse injection continuity type",
559  continuity);
560  std::string for_injection_continuity_type =
561  std::string{"Injection continuity type "} + continuity + " : ";
562 
563  auto time_units =
564  io::read<std::string>(split_line, param_index,
565  in_file + for_injection_continuity_type +
566  "Could not parse injection time units");
567  if (!TimeUnits{}.contains(time_units))
568  throw std::runtime_error{in_file + for_injection_continuity_type +
569  "Time units " + time_units + " not supported"};
570  double time_unit_factor =
571  ptof::time_unit_factor(time_units, params_transport, params_reaction);
572 
573  auto injection_time_start =
574  io::read<double>(split_line, param_index,
575  in_file + for_injection_continuity_type +
576  "Could not parse injection start time");
577  injection_time_start *= time_unit_factor;
578 
579  if (continuity == "pulse") {
580  if (type != InitialConditionList::Type::prescribed_positions_masses &&
581  type !=
582  InitialConditionList::Type::prescribed_positions_masses_tags) {
583  return std::make_unique<InjectionParameters_Pulse>(
584  injection_time_start,
585  io::read<double>(split_line, param_index,
586  in_file + for_injection_continuity_type +
587  "Could not parse total mass"));
588  }
589  return std::make_unique<InjectionParameters_PulseNoMass>(
590  injection_time_start);
591  } else if (continuity == "continuous") {
592  auto injection_time_end =
593  io::read<double>(split_line, param_index,
594  in_file + for_injection_continuity_type +
595  "Could not parse injection end time");
596  injection_time_end *= time_unit_factor;
597  auto discretization_option = io::read<std::string>(
598  split_line, param_index,
599  in_file + for_injection_continuity_type +
600  "Could not parse time discretization option");
601  double injection_time_step;
602  if (discretization_option == "time_step") {
603  io::read(split_line, param_index,
604  in_file + for_injection_continuity_type +
605  "Could not parse discretization time step",
606  injection_time_step);
607  injection_time_step *= time_unit_factor;
608  } else if (discretization_option == "accuracy") {
609  double time_step_adv, time_step_diff, time_step_react;
610  io::read(split_line, param_index,
611  in_file + for_injection_continuity_type +
612  "Could not parse time step in units of advective, "
613  "diffusive, and reactive time",
614  time_step_adv, time_step_diff, time_step_react);
615  injection_time_step =
616  std::min({time_step_adv * params_transport.advection_time,
617  time_step_diff * params_transport.diffusion_time,

36 template <typename State>
37 void boundary_reflecting(State &state, Foam::point const &contact_point,
38  Foam::vector const &unit_normal) {
39  auto leftover_jump = make_point(state.position) - contact_point;
40 
41  // When a particle starts at a reflecting wall,
42  // it must not be reflected if it jumps towards the interior
43  // and not through the boundary
44  // This dot product can only be positive in that case
45  auto leftover_dot_normal = leftover_jump & unit_normal;
46  state.set_position(contact_point + leftover_jump +
47  (leftover_dot_normal < 0.
48  ? -2. * leftover_dot_normal * unit_normal
49  : Foam::zero{}));
50 }

68 inline auto get_boundary_conditions(std::string const &filename,
69  std::string const &delims = "\t,| ") {
70  std::map<std::string, std::string> boundary_conditions;
71  std::string comment_sequence = "#";
72 
73  auto file = useful::open_read(filename);
74  std::string line;
75  while (std::getline(file, line)) {
76  if (line.find(comment_sequence) == 0)
77  continue;
78  useful::remove_carriage_return_in_place(line);
79  useful::clear_escape_in_place(line, comment_sequence);
80  auto split_line = useful::split(line, "\t,| ");
81  if (split_line.size() != 2)
82  throw useful::parse_error(filename, line);
83  boundary_conditions[split_line[0]] = split_line[1];
84  }
85  file.close();
86 
87  return boundary_conditions;
88 }

216 template <typename State, typename Intersection, typename Boundary,
217  typename Locator>
218 auto periodic_intersection(State state_outside, State const &state_image,
219  Intersection const &intersection,
220  Boundary const &boundary, Locator const &locator) {
221  // Compute displacement from intersection to outside position.
222  auto displacement = make_point(state_outside.position) - intersection.point();
223 
224  // Slight offset from intersection towards the outside point.
225  // The goal is to find the periodic image of a point close to the
226  // intersection, so we can find its periodic image within the domain and then
227  // use it to find the periodic image of the intersection itself.
228  auto offset_pos = offset_forward_face(
229  intersection.point(), intersection.index(), displacement, locator);
230  auto offset = offset_pos - intersection.point();
231 
232  // Ensure offset position is closer to intersection than outside position.
233  // Otherwise, use displacement.
234  if (Foam::magSqr(offset) < Foam::magSqr(displacement))
235  state_outside.set_position(offset_pos);
236  else {
237  offset = make_point(state_outside.position) - intersection.point();
238  state_outside.set_position(intersection.point() + offset);
239  }
240 
241  // Place the offset point at its periodic image
242  boundary(state_outside);
243 
244  // Find the periodic image of the intersection by looking from outside the
245  // domain towards the offset point
246 
247  // For some meshes, numerical error can lead to the intersection point not
248  // being found. Avoid breaking by repeatedly doubling the size of the interval
249  // until the intersection is found or too many tries have been made.
250  std::size_t max_nr_doublings = 20;
251  std::size_t iter = 0;
252  do {
253  offset *= 2;
254  auto new_intersection = locator.mesh_search().intersection(
255  make_point(state_outside.position) - offset,
256  make_point(state_outside.position));
257  if (new_intersection.hit())
258  return new_intersection;
259  } while (iter++ < max_nr_doublings);
260 
261  return Intersection{};
262 }

358 template <typename BoundaryCondition, typename Mesh>
359 auto extract_cartesian_periodic_boundaries(
360  BoundaryCondition const &boundary_conditions, Mesh const &mesh,
361  std::size_t dim) {
362  std::vector<std::vector<double>> boundaries_dim(dim);
363  for (auto const &bc : boundary_conditions)
364  if (bc.second == "periodic") {
365  // Find average and variance of patch face centers.
366  auto const &patch_id = mesh.boundaryMesh().findPatchID(bc.first, false);
367  auto const &patch = mesh.boundaryMesh()[patch_id];
368  auto face_start = patch.start();
369  Foam::point average_face_center = Foam::point::zero;
370  Foam::point variance_face_center = Foam::point::zero;
371  Foam::scalar total_area = 0.;
372  for (auto face = face_start; face < face_start + patch.size(); ++face) {
373  auto area = face_area(face, mesh);
374  auto center = face_center(face, mesh);
375  auto weighted_center = area * center;
376  total_area += area;
377  average_face_center += weighted_center;
378  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
379  variance_face_center[dd] += weighted_center[dd] * center[dd];
380  }
381  average_face_center /= total_area;
382  variance_face_center /= total_area;
383  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
384  variance_face_center[dd] -=
385  average_face_center[dd] * average_face_center[dd];
386 
387  // Choose dimension with least variance.
388  auto it_min = std::min_element(variance_face_center.begin(),
389  variance_face_center.end());
390  std::size_t dd = std::distance(variance_face_center.begin(), it_min);
391  boundaries_dim[dd].push_back(average_face_center[dd]);
392  }
393 
394  // Check which dimensions have periodic BCs.
395  // If a dimension has periodic BCs they must be exactly two.
396  std::vector<bool> dim_has_periodic_bcs(dim, 0);
397  for (std::size_t dd = 0; dd < dim; ++dd)
398  if (boundaries_dim[dd].size() != 0) {
399  if (boundaries_dim[dd].size() != 2)
400  throw std::runtime_error{"Extracted " +
401  std::to_string(boundaries_dim[dd].size()) +
402  " periodic boundaries "
403  "along cartesian dimension " +
404  std::to_string(dd)};
405  dim_has_periodic_bcs[dd] = 1;
406  }
407 
408  // Periodic conditions are checked in order across Cartesian dimensions.
409  // If there are periodic BCs in all dimensions starting from 0 up to some
410  // dimension, add only those boundaries If there are gap dimensions between
411  // periodic dimensions, add {-inf, inf} boundaries.
412  bool found_nonperiodic = 0;
413  bool nonconsecutive_dims = 0;
414  for (bool periodic : dim_has_periodic_bcs) {
415  if (found_nonperiodic && periodic) {
416  nonconsecutive_dims = 1;
417  break;
418  }
419  if (!periodic)
420  found_nonperiodic = 1;
421  }
422  std::vector<std::pair<double, double>> boundaries;
423  boundaries.reserve(boundaries_dim.size());
424  for (std::size_t dd = 0; dd < boundaries_dim.size(); ++dd) {
425  if (dim_has_periodic_bcs[dd]) {
426  boundaries.push_back(
427  {std::min(boundaries_dim[dd][0], boundaries_dim[dd][1]),
428  std::max(boundaries_dim[dd][0], boundaries_dim[dd][1])});
429  } else if (nonconsecutive_dims) {
430  boundaries.push_back({-std::numeric_limits<double>::infinity(),
431  std::numeric_limits<double>::infinity()});
432  }
433  }
434 
435  return boundaries;
436 }

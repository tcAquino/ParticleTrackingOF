38 template <typename State>
39 void boundary_reflecting(State &state, Foam::point const &contact_point,
40  Foam::vector const &unit_normal) {
41  auto leftover_jump = make_point(state.position) - contact_point;
42 
43  // When a particle starts at a reflecting wall,
44  // it must not be reflected if it jumps towards the interior
45  // and not through the boundary
46  // This dot product can only be positive in that case
47  auto leftover_dot_normal = leftover_jump & unit_normal;
48  state.set_position(contact_point + leftover_jump +
49  (leftover_dot_normal < 0.
50  ? -2. * leftover_dot_normal * unit_normal
51  : Foam::zero{}));
52 }

208 template <typename State, typename Intersection, typename Boundary,
209  typename Locator>
210 auto periodic_intersection(State state_outside, State const &state_image,
211  Intersection const &intersection,
212  Boundary const &boundary, Locator const &locator) {
213  // Compute displacement from intersection to outside position.
214  auto displacement = make_point(state_outside.position) - intersection.point();
215 
216  // Slight offset from intersection towards the outside point.
217  // The goal is to find the periodic image of a point close to the
218  // intersection, so we can find its periodic image within the domain and then
219  // use it to find the periodic image of the intersection itself.
220  auto offset_pos = offset_forward_face(
221  intersection.point(), intersection.index(), displacement, locator);
222  auto offset = offset_pos - intersection.point();
223 
224  // Ensure offset position is closer to intersection than outside position.
225  // Otherwise, use displacement.
226  if (Foam::magSqr(offset) < Foam::magSqr(displacement))
227  state_outside.set_position(offset_pos);
228  else {
229  offset = make_point(state_outside.position) - intersection.point();
230  state_outside.set_position(intersection.point() + offset);
231  }
232 
233  // Place the offset point at its periodic image
234  boundary(state_outside);
235 
236  // Find the periodic image of the intersection by looking from outside the
237  // domain towards the offset point
238 
239  // For some meshes, numerical error can lead to the intersection point not
240  // being found. Avoid breaking by repeatedly doubling the size of the interval
241  // until the intersection is found or too many tries have been made.
242  std::size_t max_nr_doublings = 20;
243  std::size_t iter = 0;
244  do {
245  offset *= 2;
246  auto new_intersection = locator.mesh_search().intersection(
247  make_point(state_outside.position) - offset,
248  make_point(state_outside.position));
249  if (new_intersection.hit())
250  return new_intersection;
251  } while (iter++ < max_nr_doublings);
252 
253  return Intersection{};
254 }

365 template <typename BoundaryCondition, typename Mesh>
366 auto extract_cartesian_periodic_boundaries(
367  BoundaryCondition const &boundary_conditions, Mesh const &mesh,
368  std::size_t dim) {
369  std::vector<std::vector<double>> boundaries_dim(dim);
370  for (auto const &bc : boundary_conditions)
371  if (bc.second == "periodic") {
372  // Find average and variance of patch face centers.
373  auto const &patch_id = mesh.boundaryMesh().findPatchID(bc.first, false);
374  auto const &patch = mesh.boundaryMesh()[patch_id];
375  auto face_start = patch.start();
376  Foam::point average_face_center = Foam::point::zero;
377  Foam::point variance_face_center = Foam::point::zero;
378  Foam::scalar total_area = 0.;
379  for (auto face = face_start; face < face_start + patch.size(); ++face) {
380  auto area = face_area(face, mesh);
381  auto center = face_center(face, mesh);
382  auto weighted_center = area * center;
383  total_area += area;
384  average_face_center += weighted_center;
385  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
386  variance_face_center[dd] += weighted_center[dd] * center[dd];
387  }
388  average_face_center /= total_area;
389  variance_face_center /= total_area;
390  for (std::size_t dd = 0; dd < variance_face_center.size(); ++dd)
391  variance_face_center[dd] -=
392  average_face_center[dd] * average_face_center[dd];
393 
394  // Choose dimension with least variance.
395  auto it_min = std::min_element(variance_face_center.begin(),
396  variance_face_center.end());
397  std::size_t dd = std::distance(variance_face_center.begin(), it_min);
398  boundaries_dim[dd].push_back(average_face_center[dd]);
399  }
400 
401  // Check which dimensions have periodic BCs.
402  // If a dimension has periodic BCs they must be exactly two.
403  std::vector<bool> dim_has_periodic_bcs(dim, 0);
404  for (std::size_t dd = 0; dd < dim; ++dd)
405  if (boundaries_dim[dd].size() != 0) {
406  if (boundaries_dim[dd].size() != 2)
407  throw std::runtime_error{"Extracted " +
408  std::to_string(boundaries_dim[dd].size()) +
409  " periodic boundaries "
410  "along cartesian dimension " +
411  std::to_string(dd)};
412  dim_has_periodic_bcs[dd] = 1;
413  }
414 
415  // Periodic conditions are checked in order across Cartesian dimensions.
416  // If there are periodic BCs in all dimensions starting from 0 up to some
417  // dimension, add only those boundaries If there are gap dimensions between
418  // periodic dimensions, add {-inf, inf} boundaries.
419  bool found_nonperiodic = 0;
420  bool nonconsecutive_dims = 0;
421  for (bool periodic : dim_has_periodic_bcs) {
422  if (found_nonperiodic && periodic) {
423  nonconsecutive_dims = 1;
424  break;
425  }
426  if (!periodic)
427  found_nonperiodic = 1;
428  }
429  std::vector<std::pair<double, double>> boundaries;
430  boundaries.reserve(boundaries_dim.size());
431  for (std::size_t dd = 0; dd < boundaries_dim.size(); ++dd) {
432  if (dim_has_periodic_bcs[dd]) {
433  boundaries.push_back(
434  {std::min(boundaries_dim[dd][0], boundaries_dim[dd][1]),
435  std::max(boundaries_dim[dd][0], boundaries_dim[dd][1])});
436  } else if (nonconsecutive_dims) {
437  boundaries.push_back({-std::numeric_limits<double>::infinity(),
438  std::numeric_limits<double>::infinity()});
439  }
440  }
441 
442  return boundaries;
443 }
